90 void PolylineDiscretisation::computeIntersectedGridCellWithCurrentSegment()
91 {
92  m_intersected_grid_cells.clear();
93 
94  Real3 polylineCellBFL;

96  math::min(m_segment_cur.getP1Y(), m_segment_cur.getP2Y()),

98 
99  Real3 polylineCellTBR;
100 
101  polylineCellTBR.assign(math::max(m_segment_cur.getP1X(), m_segment_cur.getP2X()),
102  math::max(m_segment_cur.getP1Y(), m_segment_cur.getP2Y()),
103  math::max(m_segment_cur.getP1Z(), m_segment_cur.getP2Z()));
104 
105  IntegerSharedArray polylineIdxCellBFL(3);
106  m_grid->getIJK(polylineCellBFL, polylineIdxCellBFL);
107  IntegerSharedArray polylineIdxCellTBR(3);
108  m_grid->getIJK(polylineCellTBR, polylineIdxCellTBR);
109 
110  Integer iInfP = math::min(math::max(polylineIdxCellBFL[0], 0), m_grid->getNbX() - 1);
111  Integer iSupP = math::min(math::max(polylineIdxCellTBR[0], 0), m_grid->getNbX() - 1);
112  Integer jInfP = math::min(math::max(polylineIdxCellBFL[1], 0), m_grid->getNbY() - 1);
113  Integer jSupP = math::min(math::max(polylineIdxCellTBR[1], 0), m_grid->getNbY() - 1);
114  Integer kInfP = math::min(math::max(polylineIdxCellBFL[2], 0), m_grid->getNbZ() - 1);
115  Integer kSupP = math::min(math::max(polylineIdxCellTBR[2], 0), m_grid->getNbZ() - 1);
116 
117  for (Integer k = kInfP; k <= kSupP; k++)
118  {
119  for (Integer j = jInfP; j <= jSupP; j++)
120  {
121  for (Integer i = iInfP; i <= iSupP; i++)
122  {
123  m_intersected_grid_cells.add(m_grid->getIJK(i, j, k));
124  }
125  }

200  {

202  SharedArray<Real3> coords(nb_nodes+1) ;
203 
204  // compute polygon centroid
205  for(Integer i=0;i<nb_nodes;++i)
206  {

208  fcenter += coords[i] ;
209  }
210  // triangular geometric decomposition of the polygon

212  coords[nb_nodes] = coords[0] ;
213  for(Integer j=0;j<nb_nodes;++j)
214  {
215  computeIntersection(coords[j], coords[j+1], fcenter);
216  coordonnee = m_inter.getIntersectionValue(0);
217  if (m_inter.getNbIntersection() > 0 && m_coord_found.contains(coordonnee) == false)
218  {

220  }
221  }
