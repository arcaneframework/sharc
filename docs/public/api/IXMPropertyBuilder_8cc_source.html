399 void
400 ArcGeoSim::IXMPropertyBuilderImpl::
401 _checkDateEventConformity(const IXMPropertyIncrementDataList& properties, const Real& event_time)
402 {
403  // Date Event conformity :
404  // x- at most one values tag (may not have)
405  // x- if the values tag exist its time attribute must be equal to event_time
406 
407  m_is_event_conform = true;
408 
409  // Check time attribute values
410  Integer nb_false_property = 0;
411  for (IXMPropertyIncrementDataList::const_iterator iProperty = properties.begin(); iProperty != properties.end(); ++iProperty)
412  {
413  String name = iProperty->name;
414  m_is_property_conform[name] = true;
415  Integer values_tag_number = iProperty->support_and_values.item_infos.size();
416  if ( values_tag_number > 1)
417  {
418  m_is_property_conform[name] = false;
419  ++nb_false_property;
420  }
421  else if (values_tag_number == 1)
422  {
423  if (iProperty->support_and_values.item_infos.front().time != event_time)
424  {
425  m_is_property_conform[name] = false;
426  ++nb_false_property;
427  }
428  }
429  }
430  // Save current event times
431  m_event_begin_time = event_time;
432  m_event_end_time = event_time;
433  // Event is not conformed if all the properties are not conformed
434  Arcane::Integer property_size = properties.size();
435  if ( property_size > 0 && (nb_false_property == property_size)) m_is_event_conform = false;
436 }

440 void
441 ArcGeoSim::IXMPropertyBuilderImpl::
442 _checkPeriodEventConformity(const IXMPropertyIncrementDataList& properties, const Real& event_begin_time,const Real& event_end_time)
443 {
444  // Period Event Conformity :
445  // x- at most 2 values tags (0 and 1 are OK)
446  // x- time attribute values must belong to {event_begin_time,event_end_time}
447  // x- the two tags with the same time value is forbidden :
448  // i.e one must have (first_time == begin_time and second_time == end_time) or (first_time = end_time and second_time = begin)
449  // x- non Real based values (Real or Real3) for interpolation
450 
451  m_is_event_conform = true;
452 
453  // Check time attribute values
454  Integer nb_false_property = 0;
455  for (IXMPropertyIncrementDataList::const_iterator iProperty = properties.begin(); iProperty != properties.end(); ++iProperty)
456  {
457  m_is_property_conform[iProperty->name] = true;
458  Integer values_tag_number = iProperty->support_and_values.item_infos.size();
459  if (values_tag_number > 2)
460  {
461  m_is_property_conform[iProperty->name] = false;
462  ++nb_false_property;
463  }
464  else if (values_tag_number == 1)
465  {
466  Real time = iProperty->support_and_values.item_infos.front().time;
467  if (time != event_begin_time && time != event_end_time)
468  {
469  m_is_property_conform[iProperty->name] = false ;
470  ++nb_false_property;
471  }
472  else if (time == event_end_time &&
473  (ArcGeoSim::DataTypeConversion::basicDataType(ArcGeoSim::IXMTools::dataTypeFromIXMName(iProperty->data_type)) != DT_Real) ) // Non real based values for interpolation
474  {
475  m_is_property_conform[iProperty->name] = false ;
476  ++nb_false_property;
477  }
478  }
479  else if (values_tag_number == 2)
480  {
481  Real first_time = iProperty->support_and_values.item_infos.front().time;
482  Real second_time = iProperty->support_and_values.item_infos.back().time;
483  if ( (first_time != event_begin_time || second_time != event_end_time) &&
484  (first_time != event_end_time || second_time != event_begin_time))
485  {
486  m_is_property_conform[iProperty->name] = false;
487  ++nb_false_property;
488  }
489  else if (ArcGeoSim::DataTypeConversion::basicDataType(ArcGeoSim::IXMTools::dataTypeFromIXMName(iProperty->data_type)) != DT_Real) // Non real based values for interpolation
490  {
491  m_is_property_conform[iProperty->name] = false;
492  ++nb_false_property;
493  }
494  }
495  }
496 
497  // Save current event times
498  m_event_begin_time = event_begin_time;
499  m_event_end_time = event_end_time;
500  // Event is not conformed if all the properties are not conformed
501  Arcane::Integer property_size = properties.size();
502  if ( property_size > 0 && (nb_false_property == property_size)) m_is_event_conform = false;
503 }

535 {

537  if (filled_item_ivariable->itemKind() == Arcane::IK_Unknown) return; // Global variable case

649  // Construct the ItemVariableScalarRefT<Byte> with all elements to false
650  // Get property IVariable (used to have item kind)
651  ARCANE_ASSERT((m_variable_mng),(String::format("Cannot determine if Property {0} does exist. Cannot give filled item information.",m_property_name).localstr()));
652  IVariable* property_variable = m_variable_mng->findMeshVariable(m_variable_mng->subDomain()->defaultMesh(),m_property_name);
653  ARCANE_ASSERT((property_variable),(String::format("Property {0} does not exist",m_property_name).localstr()));

655  ARCANE_ASSERT((property_variable->itemKind() != Arcane::IK_Unknown),(String::format("Property {0} is not defined on mesh. Cannot give filled item indicator variable",
656  m_property_name).localstr()));
657  // Create item variable
658  String variable_name = PropertyDescription::modifiedItemsVariableName(m_property_name);
659  Arcane::VariableItemByte filled_item_variable(VariableBuildInfo(m_variable_mng->subDomain()->defaultMesh(),variable_name,IVariable::PPersistant),property_variable->itemKind());
660  filled_item_variable.fill(false);
661  return filled_item_variable;
662  }
663  else
664  {
665  ARCANE_ASSERT((m_filled_item_variable),(Arcane::String::format("IVariable indicating property {0} changes is NULL.",
666  m_property_name).localstr()));
667  ARCANE_ASSERT((m_filled_item_variable->itemKind() != Arcane::IK_Unknown),(String::format("Property {0} is not defined on mesh. Cannot give a modified item indicator variable",
668  m_property_name).localstr()));

671  }

959 void
960 ArcGeoSim::IXMPropertyBuilderImpl::PropertyDescription::
961 _loadMeshVariable(const bool check_property_existence) noexcept(false)
962 {
963  // Get property IVariable
964  IVariable* variable = this->m_mesh->subDomain()->variableMng()->findMeshVariable(this->m_mesh, m_name);
965 
966  // Store item group
967  m_group = this->m_mesh->findGroup(m_group_name);
968  m_trace_mng->info()<<"PropertyDescription:: GROUP SIZE("<<m_group_name<<")="<<m_group.size();
969  if (m_group.null()) throw InconsistentPropertyDescriptionException(m_name,Arcane::String::format("property group {0} does not exist",m_group_name));
970  m_trace_mng->debug() << Arcane::String::format("Property {0} group name : {1}", m_name, m_group_name);
971 
972  // Check property existence
973  if (!variable)
974  if (check_property_existence) throw SkippedPropertyException(m_name, "property variable does not exist.");
975  else // Create variable
976  {
977  // Properties are created on ItemGroup (may be partial)
978  Arcane::eItemKind item_kind = itemKind();
979  // is Partial ?
980  bool is_partial = (!m_group.isAllItems());
981 
982 #if (ARCANE_VERSION >= 30003)
983  if(item_kind==ItemConnectionImpl::ik_type)
984  {
985  m_created_property_variable_ref = ArcGeoSim::VariableUtils::createMeshVariableRef(m_name,
986  ArcGeoSim::IXMTools::dataTypeFromIXMName(m_type),
987  IK_DoF,m_dimension,is_partial,m_group);
988  }
989  else
990  {
991  m_created_property_variable_ref = ArcGeoSim::VariableUtils::createMeshVariableRef(m_name,
992  ArcGeoSim::IXMTools::dataTypeFromIXMName(m_type),
993  item_kind,m_dimension,is_partial,m_group);
994  }
995 #else
996  m_created_property_variable_ref = ArcGeoSim::VariableUtils::createMeshVariableRef(m_name,
997  ArcGeoSim::IXMTools::dataTypeFromIXMName(m_type),
998  item_kind,m_dimension,is_partial,m_group);
999 #endif
1000  // Handle array property
1001  if (m_dimension == 2) ArcGeoSim::VariableUtils::resizeMeshVariableArray(m_created_property_variable_ref->variable(),m_array_size);
1002 
1003  // Property manager keeps a reference on the variable during its whole life to ensure variable is not destroyed
1004  m_property_variable = m_created_property_variable_ref->variable();
1005  }
1006  else
1007  {
1008  // Check variable group if existing variable is partial
1009  if (variable->isPartial())
1010  {
1011  if (variable->itemGroup() != m_group) throw InconsistentPropertyDescriptionException(m_name,
1012  Arcane::String::format("property variable exist but is defined on {0} while ixm mesh file defines it on {1}",
1013  variable->itemGroup().name(),m_group_name));
1014  }
1015  // Store the property IVariable
1016  variable->setUsed(true); // to handle property declared in axl and not yet allocated
1017  // Handle array property if not sized (possible if the property was only declared in axl
1018  if (m_dimension == 2) ArcGeoSim::VariableUtils::resizeMeshVariableArray(variable,m_array_size);
1019  m_property_variable = variable;
1020  }
1021 }

1063 void
1064 ArcGeoSim::IXMPropertyBuilderImpl::PropertyDescription::
1065 _checkPropertyDescription() const noexcept(false)
1066 {
1067  // Handle Integer string case
1068  String type = m_type;
1069 
1070  // Check data type
1071  if (type.empty()) throw InconsistentPropertyDescriptionException(m_name,"property has no data type");
1072  // Check data type, item kind, skip property if not ok
1073  if (ArcGeoSim::IXMTools::dataTypeFromIXMName(type) != m_property_variable->dataType()) throw InconsistentPropertyDescriptionException(
1074  m_name,
1075  String::format("property is declared in mesh to be a {0} variable while it is a {1} variable",
1076  type, ArcGeoSim::IXMTools::dataTypeIXMName(m_property_variable->dataType())));
1077 
1078  if (m_kind != ArcGeoSim::IXMTools::itemKindIXMName(m_property_variable->itemKind())) throw InconsistentPropertyDescriptionException(
1079  m_name,
1080  String::format("property is declared in mesh to be a {0} variable while it is a {1} variable.",
1081  m_kind, ArcGeoSim::IXMTools::itemKindIXMName(m_property_variable->itemKind())) );
1082 
1083  // Check array size
1084  if (m_dimension == 2) // mesh array variable
1085  {
1086  Arcane::Integer item_number, array_size;
1087  ArcGeoSim::VariableUtils::meshVariableArraySize(m_property_variable->data(),item_number,array_size);
1088  if (array_size != m_array_size) throw InconsistentPropertyDescriptionException(
1089  m_name,
1090  String::format("property is an existing mesh array variable of size {0} but is declared as a mesh array variable of size {1}",
1091  array_size, m_array_size));
1092  }
1093  else if (m_dimension == 1 && m_property_variable->itemKind() == Arcane::IK_Unknown) // array variable (not on mesh)
1094  {
1095  if (m_array_size != m_property_variable->nbElement())
1096  throw InconsistentPropertyDescriptionException(m_name, String::format(" global property size is {0} while we expected {1}",
1097  m_property_variable->nbElement(),
1098  m_array_size));
1099  }
1100  else if ( (m_dimension == 1 && m_property_variable->itemKind() != Arcane::IK_Unknown) && m_array_size>1 ) // mesh scalar variable
1101  {
1102  throw InconsistentPropertyDescriptionException(m_name, String::format(" property is scalar on mesh and array size is not equal to 1 but to {0}",
1103  m_array_size));
1104  }
1105  else if (m_dimension == 0 && m_array_size > 1) // scalar variable (not on mesh)
1106  {
1107  throw InconsistentPropertyDescriptionException(m_name, String::format(" property is a scalar array size is not equal to 1 but to {0}",
1108  m_array_size));
1109 
1110  }

1116 void
1117 ArcGeoSim::IXMPropertyBuilderImpl::PropertyDescription::
1118 saveModifiedItems(const ItemGroup& modified_items)
1119 {
1120  if (modified_items.null()) return;
1121 
1122  String variable_name = modifiedItemsVariableName(m_name);
1123 
1124  // Create (if needed) and fill ItemVariable<Byte> to indicate modified items
1125 
1126  // Check if modified item variable already created
1127  IVariable* filled_item_variable = this->m_mesh->subDomain()->variableMng()->findMeshVariable(this->m_mesh,variable_name);
1128 
1129  // Potential new items group
1130  ItemGroup new_items;
1131 
1132  // Create if does not exist
1133  if (!filled_item_variable)
1134  {
1135  Arcane::IItemFamily * item_family = this->itemFamily(itemKind()) ;
1136  // Create Mesh Variable builder
1137  VariableBuildInfo builder(this->m_mesh,variable_name,item_family->name(),IVariable::PPersistant);
1138  ItemVariableScalarRefT<Byte> filled_item_variable(builder,item_family->itemKind());
1139  filled_item_variable.fill(false);
1140  m_filled_item_variable = filled_item_variable.variable();
1141  }
1142  else
1143  {
1144  m_filled_item_variable = filled_item_variable;
1145  // Get new items potentially added in IEvolutiveMesh transformations
1146 #ifdef USE_EVOLUTIF
1147  if (m_evolutive_mesh) {
1148  switch (itemKind()) {
1149  case IK_Cell:
1150  new_items = m_evolutive_mesh->newCells();
1151  break;
1152  case IK_Face:
1153  new_items = m_evolutive_mesh->newFaces();
1154  break;
1155  case IK_Node:
1156  new_items = m_evolutive_mesh->newNodes();
1157  break;
1158  default:
1159  m_trace_mng->fatal() << "Unsupported item kind for evolutive mesh properties.";
1160  break;
1161  }
1162  }
1163 #endif
1164  }
1165 
1166 #ifdef USE_EVOLUTIF
1167  // Update filled item variable
1168  ItemVariableScalarRefT<Byte> modified_item_variable (m_filled_item_variable);
1169  // Put to false new items if needed. This is not redundant with what is done in updateFilledItemVariable() in a case
1170  // where the filled item variable exist and is not registered in IXMPropertyBuilder::m_property_filled
1171  // (i.e. created by the user before the corresponding property is modified).
1172  if (!new_items.null()) modified_item_variable.fill(false,new_items);
1173 
1174  // Put to true modified items
1175  modified_item_variable.fill(true,modified_items);
1176 #endif
1177 }
