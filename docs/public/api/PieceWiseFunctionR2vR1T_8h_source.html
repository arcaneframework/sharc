463 template<typename T>
464  void
465  PieceWiseFunctionR2vR1T<T>::addTableVector(const RealArray & x0,
466  const Real & x1, const RealArray & y, const VectorType::eVectorType vectorType)
467  {
468  bool isRow = vectorType == VectorType::row;
469  Integer index = isRow ? 1: 0 ;
470 
471  // check vector values before filling it
472  if (x0.size() != y.size())
473  {
474  throw FatalErrorException(A_FUNCINFO,
475  String::format(
476  "inconsistent length between vector (={0}) and index (={1}): should be equal",
477  y.size(), x0.size()));
478  }
479  if (x1 < m_x[index][m_last_point_index[index]])
480  {
481  throw FatalErrorException(A_FUNCINFO,
482  "vector index should be strictly increasing between min and max");
483  }
484  Integer x0size = x0.size();
485  ComponentDataIterator it;
486  Integer rowSize = dim1Size(m_y);
487  Integer colSize = dim2Size(m_y);
488 
489  if (!m_filled)
490  {
491  // check vector values before filling it
492  if (! checkAxeDef(x0, vectorType))
493  {
494  throw FatalErrorException(A_FUNCINFO,
495  "indexes values should be strictly increasing between min and max");
496  }
497  //Table initialization, set axe definition m_x
498  Integer axe = isRow ? 0: 1 ;
499  it = m_x[axe].begin();
500  for (Integer i = 0; i < m_first_point_index[axe]; i++) it++;
501  m_x[axe].insert(it, x0.begin(), x0.end());
502 
503  it = m_x_scan[axe].begin();
504  for (Integer i = 0; i < m_first_point_index[axe]; i++) it++;
505  m_x_scan[axe].insert(it, x0.begin(), x0.end());
506 
507  m_inserted_point_number[axe] += x0size;
508  m_total_point_number[axe] += x0size;
509  m_last_point_index[axe] = m_first_point_index[axe] + x0size - 1;
510 
511  rowSize += isRow ? x0size: 0 ;
512  colSize += isRow ? 0: x0size ;
513  m_filled = true;
514  }
515  // resize table for the new vector y and fill it
516  rowSize += isRow ? 0: 1 ;
517  colSize += isRow ? 1: 0 ;
518  resize(m_y,rowSize,colSize);
519  // m_yview is the actual user table
520 // Matrixindice indices;

522  if (isRow)

528  else

531  for (Integer index = 0; index < y.size(); index++)
532  m_yview[m_inserted_point_number[0]][index] = y[index];
533  }
534

536  index = isRow ? 1: 0 ;
537  it = m_x[index].begin();
538  for (Integer i = 0; i < m_last_point_index[index]+1; i++) it++;
539  m_x[index].insert(it, x1);
540 
541  it = m_x_scan[index].begin();
