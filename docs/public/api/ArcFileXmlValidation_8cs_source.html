276  if (!yfound)
277  yindex = table.Count;

336  public static string GetAbsoluteXPath(this XElement element)
337  {
338  if (element == null)
339  {
340  throw new ArgumentNullException("element");
341  }
342 
343  Func<XElement, string> relativeXPath = e =>
344  {
345  int index = e.IndexPosition();
346  string name = e.Name.LocalName;
347 
348  // If the element is the root, no index is required
349 
350  return (index == -1) ? "/" + name : string.Format
351  (
352  "/{0}[{1}]",
353  name,
354  index.ToString()
355  );
356  };
357 
358  var ancestors = from e in element.Ancestors()
359  select relativeXPath(e);
360 
361  return string.Concat(ancestors.Reverse().ToArray()) +
362  relativeXPath(element);
363  }
