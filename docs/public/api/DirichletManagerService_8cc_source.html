127 void
128 DirichletManagerService::
129 init(ArcRes::System& system)
130 {
131  if(m_initialized) return;
132 
133  _checkSizeMulti();
134 
135  for(Arcane::Integer i = 0; i < options()->boundary.size(); ++i) {
136 
137  auto model_name = Arcane::String::format("Dirichlet_{0}",i);
138 
139  const Arcane::String face_group_name = options()->boundary[i].faceGroup();
140  ItemGroup group = mesh()->faceFamily()->findGroup(face_group_name);
141  if(group.null()) {
142  fatal() << "Dirichlet model cell zone "<<face_group_name<<" not exist";
143  }
144  auto name = Arcane::String::format("{0}_{1}",model_name,group.name());
145  auto* folder = new Law::PartialVariableFaceFolder(name,mesh());
146  m_boundaries.add(folder);
147  folder->setSupport(group);
148  if(options()->boundary[i]->name.isPresent()) {
149  m_boundary_names.add(options()->boundary[i]->name());
150  } else {
151  m_boundary_names.add(model_name);
152  }
153 
154  //==================
155  // Add Darcy Velocity
156  //==================
157 
158  ENUMERATE_PHASE(iphase, system.fluidSubSystem().phases()) {
159  const ArcRes::Phase & phase = *iphase;
160  folder->addVariable(ArcRes::DarcyVelocity(phase));
161  }
162 
163  //==================
164  // Space-time
165  //==================
166 
167  for(Arcane::Integer j = 0; j < options()->boundary[i].spaceTimeLimitCondition.size(); ++j) {
168  auto& condition(options()->boundary[i].spaceTimeLimitCondition[j]) ;
169  auto p = ArcRes::XPath::scalarRealProperty(system, condition.property());
170  folder->addVariable(p);
171  condition->function()->init() ;
172  }
173 
174  //==================
175  // Single
176  //==================
177 
178  for(Arcane::Integer j = 0; j < options()->boundary[i].limitCondition.size(); ++j) {
179  auto p = ArcRes::XPath::scalarRealProperty(system, options()->boundary[i].limitCondition[j].property());
180  folder->addVariable(p);
181  }
182 
183  //==================
184  // Multi
185  //==================
186 
187  for(Arcane::Integer j = 0; j < options()->boundary[i].multiLimitCondition.size(); ++j) {
188 
189  Arcane::Integer num_prop = options()->boundary[i]->multiLimitCondition[j]->property.size();
190 
191  for ( Arcane::Integer iprop = 0 ; iprop < num_prop; iprop++ )
192  {
193  auto p = ArcRes::XPath::scalarRealProperty(system, options()->boundary[i].multiLimitCondition[j]->property[iprop]);
194  folder->addVariable(p);
195  }
196  }
197  }
198 
199  _compute(system);
200 
201  m_initialized = true;
202 }

206 void
207 DirichletManagerService::
208 _compute(ArcRes::System& system)
209 {
210  ArcGeoSim::AppService<IExpressionMng> expression_mng;
211  ArcGeoSim::AppService<IGeometryMng> geometry_mng;
212 
213  IGeometryMng::Real3Variable face_centers
214  = geometry_mng->getReal3VariableProperty(allFaces(),IGeometryProperty::PCenter);
215 
216  for(Arcane::Integer i = 0; i < options()->boundary.size(); ++i) {
217 
218  auto& boundary = *(m_boundaries[i]);
219 
220  auto group = boundary.support();
221 
222  //==================
223  // Space-time
224  //==================
225  for(Arcane::Integer j = 0; j < options()->boundary[i]->spaceTimeLimitCondition.size(); ++j) {
226  auto& condition(options()->boundary[i].spaceTimeLimitCondition[j]) ;
227  auto bp = ArcRes::XPath::scalarRealProperty(system, condition->property());
228 
229  auto variable = Law::values(boundary,bp);
230  ENUMERATE_FACE(iface, group) {
231  Arcane::Real3 center = face_centers[iface];
232  variable[iface] = condition ->function()->eval(center[0],center[1],
233  center[2], m_global_time());
234  }
235  }
236 
237  //==================
238  // Single
239  //==================
240 
241  for(Arcane::Integer j = 0; j < options()->boundary[i]->limitCondition.size(); ++j) {
242  auto bp = ArcRes::XPath::scalarRealProperty(system, options()->boundary[i]->limitCondition[j]->property());
243 
244  // Creation de l'expression
245  FunctionParser parser;
246  parser.init(expression_mng, NULL, traceMng());
247  parser.parseString("(x,y,z;t)->" + options()->boundary[i]->limitCondition[j].value());
248  ExpressionBuilderR3vR1Core expression(&parser);
249  expression.setParameter("t",m_global_time());
250 
251  // Remplissage
252  auto variable = Law::values(boundary,bp);
253  ENUMERATE_FACE(iface, group) {
254  Arcane::Real3 center = face_centers[iface];
255  variable[iface] = expression.eval(center[0],center[1],center[2]);
256  }
257  }
258 
259  //==================
260  // Multi
261  //==================
262 
263  for(Arcane::Integer j = 0; j < options()->boundary[i]->multiLimitCondition.size(); ++j) {
264 
265  Arcane::Integer num_prop = options()->boundary[i]->multiLimitCondition[j]->property.size();
266 
267  for ( Arcane::Integer iprop = 0 ; iprop < num_prop; iprop++ )
268  {
269  auto bp = ArcRes::XPath::scalarRealProperty(system, options()->boundary[i]->multiLimitCondition[j]->property[iprop]);
270 
271  // Creation de l'expression
272  FunctionParser parser;
273  parser.init(expression_mng, NULL, traceMng());
274  parser.parseString("(x,y,z;t)->" + options()->boundary[i]->multiLimitCondition[j].value[iprop]);
275  ExpressionBuilderR3vR1Core expression(&parser);
276  expression.setParameter("t",m_global_time());
277 
278  // Remplissage
279  auto variable = Law::values(boundary,bp);
280  ENUMERATE_FACE(iface, group) {
281  Arcane::Real3 center = face_centers[iface];
282  variable[iface] = expression.eval(center[0],center[1],center[2]);
283  }
284  }
285  }
286  }
