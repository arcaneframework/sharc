42  static float weight(const int couche, const int in_out, const int hidden_neural){
43  return float((couche + 1)*(hidden_neural + 1))/(10 + in_out);
44  }

47  return float((couche + 1))/(10 + hidden_neural);
48  }

50  // test_model_3_2_2.onnx
51  // 3 inputs, 1 hidden layer of 2 neurons , 2 outputs
52  static void _compute_onnx_expected_results(
53  const float in1,
54  const float in2,
55  const float in3,
56  float& out1,
57  float& out2
58  )
59  {
60  float neurone_cache_0 = weight(0,0,0) * in1 + weight(0,1,0) * in2 + weight(0,2,0) * in3 + biais(0,0);
61  neurone_cache_0 = neurone_cache_0>0 ? neurone_cache_0:0; // fonction d'activation ReLu
62  float neurone_cache_1 = weight(0,0,1) * in1 + weight(0,1,1) * in2 + weight(0,2,1) * in3 + biais(0,1);
63  neurone_cache_1 = neurone_cache_1>0 ? neurone_cache_1:0; // fonction d'activation ReLu
64  out1 = weight(1,0,0) * neurone_cache_0 + weight(1,1,0) * neurone_cache_1 + biais(1,0);
65  out2 = weight(1,0,1) * neurone_cache_0 + weight(1,1,1) * neurone_cache_1 + biais(1,1);
66  }

68  // end this formula match the neural network was created

108 bool
109 ONNXInferenceTesterService::
110 _unitaryTestSignature()
111 {
112  // input
113  {
114  // input size 1
115  if(m_session->GetInputCount()!=1) return true;
116  // input name: {input_1_1, input_1_2, input_1_3} law convention
117  std::string input_name = m_session->GetInputNameAllocated(0, Ort::AllocatorWithDefaultOptions()).get();
118  if(input_name.compare("{input_1_1, input_1_2, input_1_3}") != 0) return true;
119  // shape and type input
120  Ort::TypeInfo input_type_info = m_session->GetInputTypeInfo(0);
121  auto input_tensor_info = input_type_info.GetTensorTypeAndShapeInfo();
122  // tensor 3 axis
123  if(input_tensor_info.GetDimensionsCount()!=3) return true;
124  // -1 axis 1 batch size variable, 3 on axis 3: 3 inputs
125  if(input_tensor_info.GetShape() != std::vector<int64_t>{ -1, 1, 3 } ) return true;
126  // float precision model
127  if(input_tensor_info.GetElementType()!=ONNX_TENSOR_ELEMENT_DATA_TYPE_FLOAT) return true;
128 
129  }
130  std::cout<<"onnx input type and shape ok"<<std::endl;
131  // output
132  {
133  // input size 1
134  if(m_session->GetOutputCount()!=1) return true;
135  // output name: output1
136  std::string output_name = m_session->GetOutputNameAllocated(0, Ort::AllocatorWithDefaultOptions()).get();
137  if(output_name.compare("{output_1_1, output_1_2}") != 0) return true;
138  // shape and type output
139  Ort::TypeInfo output_type_info = m_session->GetOutputTypeInfo(0);
140  auto output_tensor_info = output_type_info.GetTensorTypeAndShapeInfo();
141  if(output_tensor_info.GetDimensionsCount()!=3) return true;
142  // float precision model
143  if(output_tensor_info.GetElementType()!=ONNX_TENSOR_ELEMENT_DATA_TYPE_FLOAT) return true;
144  // 2 outputs
145  if(output_tensor_info.GetShape() != std::vector<int64_t>{ -1, 1 , 2} ) return true;
146  }
147  std::cout<<"onnx output type and shape ok"<<std::endl;
148 
149  return false;
150 }

154 bool
155 ONNXInferenceTesterService::
156 _unitaryTestInference()
157 {
158  Arcane::Integer local_data_size = 1000;
159  Arcane::Integer input_tensor_size = 3 * local_data_size;
160 
161  std::vector<float> input_local_data(input_tensor_size);
162  // random values generator for properties initialization
163  Arcane::random::MinstdRand generator;
164  Arcane::random::Uniform01<Arcane::random::MinstdRand> uniform(generator);
165  for(auto& in_val : input_local_data)
166  in_val = uniform();
167 
168  Arcane::Integer output_tensor_size = 2 * local_data_size;
169  std::vector<float> output_local_data(output_tensor_size);
170 
171  // onnx tensors:
172  Ort::MemoryInfo memory_info = Ort::MemoryInfo::CreateCpu(OrtAllocatorType::OrtArenaAllocator, OrtMemType::OrtMemTypeDefault);
173 
174  // input:
175  std::vector<Ort::Value> input_tensors;
176  Ort::TypeInfo input_type_info = m_session->GetInputTypeInfo(0);
177  auto input_tensor_info = input_type_info.GetTensorTypeAndShapeInfo();
178  std::vector<int64_t> input_dims = input_tensor_info.GetShape();
179  // batch size
180  input_dims[0]=local_data_size;
181  input_tensors.push_back(Ort::Value::CreateTensor<float>(memory_info, input_local_data.data(),
182  input_tensor_size , input_dims.data(), input_dims.size()));
183 
184  // output:
185  std::vector<Ort::Value> output_tensors;
186  Ort::TypeInfo output_type_info = m_session->GetOutputTypeInfo(0);
187  auto output_tensor_info = output_type_info.GetTensorTypeAndShapeInfo();
188  std::vector<int64_t> output_dims = output_tensor_info.GetShape();
189  // batch size
190  output_dims[0]=local_data_size;
191  output_tensors.push_back(Ort::Value::CreateTensor<float>(memory_info, output_local_data.data(),
192  output_tensor_size, output_dims.data(), output_dims.size()));
193 
194  // serving names:
195  std::string input_name = m_session->GetInputNameAllocated(0, Ort::AllocatorWithDefaultOptions()).get();
196  std::vector<const char*> input_names{input_name.c_str()};
197  std::string output_name = m_session->GetOutputNameAllocated(0, Ort::AllocatorWithDefaultOptions()).get();
198  std::vector<const char*> output_names{output_name.c_str()};
199 
200  // model inference
201  m_session->Run(Ort::RunOptions{nullptr}, input_names.data(),
202  input_tensors.data(), 1, output_names.data(),
203  output_tensors.data(), 1);
204 
205  // check results
206  int index_in, index_out = 0;
207  for(int i_sample=0; i_sample<local_data_size; i_sample++)
208  {
209  // compute reference values
210  float result_1, result_2;
211  _compute_onnx_expected_results(
212  input_local_data[index_in],input_local_data[index_in+1],input_local_data[index_in+2],
213  result_1,result_2);
214  // check results with reference values
215  const float eps = std::numeric_limits<float>::epsilon();
216  if(std::fabs(result_1 - output_local_data[index_out]) > eps) return true;
217  if(std::fabs(result_2 - output_local_data[index_out+1])> eps) return true;
218  index_in+=3;
219  index_out+=2;
220  }
221 
222  std::cout<<"onnx inference ok"<<std::endl;
223 
224  return false;
225 }
