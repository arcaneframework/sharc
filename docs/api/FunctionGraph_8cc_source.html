142 FunctionGraph::DependenciesMap
143 FunctionGraph::
144 computeDependencies(DerivativesMode mode) const
145 {
146  std::map<Property, PropertyVector> dependencies;
147  if(mode == eComposition){
148  checkCycle();
149  const VertexList sort_vertices = _sortedVertex();
150  _computeDependencies(sort_vertices, dependencies);
151  }
152  else if(mode == eDirect){
153  std::pair<VertexIterator, VertexIterator> it = boost::vertices(m_graph);
154  for(; it.first != it.second; ++it.first){
155  const FunctionPtr function = m_graph[*it.first].function;
156  const PropertyVector in_props = function->in();
157  const PropertyVector out_props = function->out();
158  for(Integer i_prop = 0; i_prop<out_props.size();i_prop ++)
159  dependencies.insert(std::make_pair(out_props[i_prop],in_props));
160  }
161  }
162  return dependencies;
163 }

207 void
208 FunctionGraph::
209 _createAddDownEdge(const VertexType& new_vertex){
210  typedef std::map<VertexType, PropertyVector> VertexPropMap;
211  const PropertyVector out = m_graph[new_vertex].function->out();
212  VertexPropMap connected_vertices_to_properties;
213  for(int i=0;i<out.size();++i){
214  const Property prop = out[i];
215  const int property_id = prop.id();
216  // calcul des proprietes connectees
217  auto it = m_free_input_property_to_function_vertices.find(property_id);
218  if(it!= m_free_input_property_to_function_vertices.end()){
219  VertexList list = it->second;
220  for(auto& vert : list){
221  connected_vertices_to_properties.insert(std::make_pair(vert,PropertyVector()));
222  connected_vertices_to_properties[vert]<<prop;
223  }
224  }
225  }
226  // creation d une arete in-out
227  for(auto& vert_to_prop : connected_vertices_to_properties){
228  PropertyVector edge_prop = vert_to_prop.second;
229  _createEdge(vert_to_prop.first, new_vertex, edge_prop);
230  // nettoyage de la map d input property
231  for(int i=0;i<edge_prop.size();++i)
232  m_free_input_property_to_function_vertices.erase(edge_prop[i].id());
233  }
234 }

238 void
239 FunctionGraph::
240 _createAddUpEdge(const VertexType& new_vertex){
241  typedef std::map<VertexType, PropertyVector> VertexPropMap;
242  const PropertyVector in = m_graph[new_vertex].function->in();
243  VertexPropMap connected_vertices_to_properties;
244  for(int i=0;i<in.size();++i){
245  const Property prop = in[i];
246  const int property_id = prop.id();
247  auto it = m_output_property_to_function_vertex.find(property_id);
248  if(it == m_output_property_to_function_vertex.end()){
249  m_free_input_property_to_function_vertices.insert(std::make_pair(property_id, VertexList()));
250  m_free_input_property_to_function_vertices[property_id].push_back(new_vertex);
251  }
252  else{
253  const VertexType vertex = it->second;
254  connected_vertices_to_properties.insert(std::make_pair(vertex,PropertyVector()));
255  connected_vertices_to_properties[vertex]<<prop;
256  }
257  }
258  // creation d une arete in-out
259  for(auto& vert_to_prop : connected_vertices_to_properties)
260  _createEdge(new_vertex, vert_to_prop.first , vert_to_prop.second);
261 }

310 void
311 FunctionGraph::
312 _deleteSubProperty(IntegerConstArrayView ids, VertexList& sort_vertices, const bool recursive) const
313 {
314  const Integer num_vertices = size();
315  std::vector<bool> is_find(num_vertices, false);
316  for(Integer i=0; i<ids.size();i++){
317  auto it = m_output_property_to_function_vertex.find(ids[i]);
318  if(it == m_output_property_to_function_vertex.end()){
319  if(m_fatal_if_unregistered)
320  m_trace->fatal() <<"Function of property id="<<ids[i]<<" is not registered or root";
321  }
322  else{
323  const auto current_vertex = it->second;
324  if(!is_find[current_vertex]){
325  is_find[current_vertex]=true;
326  if(recursive)
327  _tagUpVertex(current_vertex, is_find);
328  }
329  }
330  }
331  // suppression des noeud tagges
332  for(Integer i = 0; i<num_vertices;i++){
333  if(!is_find[i])
334  sort_vertices.remove(i);
335  }
336 }

353 void
354 FunctionGraph::
355 _dependencies(const VertexType& parent_vertex, DependenciesMap& dependencies, PropertyVector& in_dependencies) const {
356  const FunctionPtr func = m_graph[parent_vertex].function;
357  const PropertyVector in_props_function = func->in();
358  for(Integer j = 0; j<in_props_function.size(); j++){
359  const Property prop = in_props_function[j];
360  bool current_prop_is_root = true;
361  std::pair<InEdgeIterator, InEdgeIterator> it = boost::in_edges(parent_vertex, m_graph);
362  for(; it.first != it.second; ++it.first){
363  EdgeType current_edge = *it.first;
364  PropertyVector in_props_edge = m_graph[current_edge].linked_properties;
365  if(in_props_edge.contains(prop) && dependencies[prop].size()>0){
366  current_prop_is_root = false;
367  break;
368  }
369  }
370  // root prop
371  if(current_prop_is_root)
372  in_dependencies<<prop;
373  else
374  in_dependencies<<dependencies[prop];
375  }
376 }
