92  static void printDotGraph(const FunctionVector& functions, std::ostream& nout)
93  {
94  PrinterGraphType graph_printer;
95  // construction du graphe
96  std::map<int, PrinterVertexType> prop_uid_to_vertex;
97  bool is_linked = false;
98  for(Arcane::Integer i_func = 0 ; i_func < functions.size() ; ++i_func) {
99  FunctionPtr func = functions[i_func];
100  const std::string name_law = func->name().localstr();
101  const PrinterVertexType vertex_func = boost::add_vertex(name_law, graph_printer);
102  get(boost::vertex_index1, graph_printer)[vertex_func] = 0;
103  const PropertyVector in_prop_vect = func->in();
104  for(Integer i = 0; i<in_prop_vect.size(); ++i){
105  PrinterVertexType vertex_prop = _findOrAddVertexProperty(in_prop_vect[i], prop_uid_to_vertex, graph_printer);
106  PrinterEdgeType new_edge ;
107  const std::string name_edge = "in " + std::to_string(i);
108  boost::tie(new_edge, is_linked) = boost::add_edge(vertex_prop, vertex_func, name_edge, graph_printer);
109  }
110  const PropertyVector out_prop_vect = func->out();
111  for(Integer i = 0; i<out_prop_vect.size(); ++i){
112  PrinterVertexType vertex_prop = _findOrAddVertexProperty(out_prop_vect[i], prop_uid_to_vertex, graph_printer);
113  PrinterEdgeType new_edge ;
114  const std::string name_edge = "out " + std::to_string(i);
115  boost::tie(new_edge, is_linked) = boost::add_edge(vertex_func, vertex_prop, name_edge, graph_printer);
116  }
117  }
118  // Ecriture du graphe
119  boost::write_graphviz(nout,graph_printer,
120  make_vertex_writer(boost::get(boost::vertex_name, graph_printer),boost::get(boost::vertex_index1, graph_printer)),
121  boost::make_label_writer(boost::get(boost::edge_name, graph_printer)));
122  };

127  static PrinterVertexType _findOrAddVertexProperty(const Property& prop,
128  std::map<int, PrinterVertexType>& prop_uid_to_vertex, PrinterGraphType& graph_printer)
129  {
130  const int prop_id = prop.id();
131  PrinterVertexType vertex_prop;
132  if(prop_uid_to_vertex.find(prop_id)==prop_uid_to_vertex.end()){
133  const std::string name =prop.name().localstr();

135  get(boost::vertex_index1, graph_printer)[vertex_prop] = 1;
136  prop_uid_to_vertex.insert(std::make_pair(prop_id, vertex_prop));
137  }
138  else{
139  vertex_prop = prop_uid_to_vertex[prop_id];
140  }
141  return vertex_prop;
