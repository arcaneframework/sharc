94 IXMPropertyInfo* IXMPropertyInfo::
95 create(IVariable* var)
96 {
97  IXMPropertyInfo* prop_info = 0;
98  switch (var->dataType()) {
99  case DT_Real:
100  prop_info = new IxmPropertyInfoT<Real>(var);
101  break;
102  case DT_Real2:
103  prop_info = new IxmPropertyInfoT<Real2>(var);
104  break;
105  case DT_Real2x2:
106  prop_info = new IxmPropertyInfoT<Real2x2>(var);
107  break;
108  case DT_Real3:
109  prop_info = new IxmPropertyInfoT<Real3>(var);
110  break;
111  case DT_Real3x3:
112  prop_info = new IxmPropertyInfoT<Real3x3>(var);
113  break;
114  case DT_Byte:
115  prop_info = new IxmPropertyInfoT<Byte>(var);
116  break;
117  case DT_Int32:
118  prop_info = new IxmPropertyInfoT<Int32>(var);
119  break;
120  case DT_Int64:
121  prop_info = new IxmPropertyInfoT<Int64>(var);
122  break;
123  case DT_String:
124  throw FatalErrorException(A_FUNCINFO,"Bad variable type");
125  break;
126  case DT_Unknown:
127  throw FatalErrorException(A_FUNCINFO,"Bad variable type");
128  break;
129  }
130  return prop_info;
131 }

145 template<typename DataType> void
146 IxmPropertyInfoT<DataType>::
147 writeProperty(XmlNode current_node, const String& format, IXM4FileInfo & fileinfo, const bool write_only_active_cells, const Real& time)
148 {
149 
150  ItemGroup group;
151 
152  if (write_only_active_cells)
153  {
154  if ( m_variable->itemKind() == Arcane::IK_Cell )
155  group = m_variable->itemGroup().activeCellGroup();
156  else
157  group = m_variable->itemGroup().own();
158  }
159  else
160  group = m_variable->itemGroup().own();
161 
162  //Arcane::eItemKind m_variable->itemKind;
163 
164 
165  // Get values arrays
166  ISubDomain* subdomain = m_variable->subDomain();
167  IParallelMng* pm = subdomain->parallelMng();
168 
169  // Create a view on variable (to handle in the same way any VariableType)
170  ArcGeoSim::VariableUtils::VariableView variable_view(m_variable);
171 
172  // Extract values on the variable item group (requires to create a GenericArray and an ArrayMng)
173  ArcGeoSim::ArrayMng* array_mng = new ArcGeoSim::ArrayMngT<DataType>(subdomain);
174  ArcGeoSim::VariableUtils::GenericArray generic_values(array_mng);
175 
176  generic_values = variable_view[group];
177 
178  // Copy extracted values in a standard Arcane::Array
179  SharedArray<DataType> values(generic_values);
180 
181  // Get uids array (or abstract_ids if they exist)
182  Int64SharedArray uids(group.size());
183  ArcGeoSim::IXMLidToUidMng lid2uid(m_variable->mesh());
184  lid2uid.itemsUniqueId(uids,group.view(),m_variable->itemKind());
185 
186  Integer array_size;
187  _getArraySize(array_size);
188 
189  // Write
190  if (group.null()) { // global variable
191  if (pm->isMasterIO()) staticBuild(current_node,format,time,values,array_size); // suppose variable was synchronized
192  }
193  else {
194  _gatherData(pm,values,uids);
195  if (pm->isMasterIO()) staticBuild(current_node,format,time,uids,values,fileinfo,array_size);
196  }
197 
198  delete array_mng;
199 }

204 template<typename DataType> void
205 IxmPropertyInfoT<DataType>::
206 _getArraySize(Arcane::Integer& array_size)
207 {
208  array_size = -1; // Dans ce cas on a une variable de type scalaire
209  // Rm: une variable de type tableau peut ��tre de taille 0, 1 ou plus
210  // array_size >= 0 ssi la variable est de type tableau
211  if (m_variable->itemGroup().null()) {// global variable
212  if (m_variable->dimension() == 1) {
213  array_size = m_variable->nbElement();
214  }
215  }
216  else {// mesh variable
217  if (m_variable->dimension() == 2) {
218  Integer nb_items;
219  ArcGeoSim::VariableUtils::meshVariableArraySize(m_variable->data(),nb_items,array_size);
220  }
221  }
222 }
