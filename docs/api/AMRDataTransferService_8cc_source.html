151 void AMRDataTransferService::init()
152 {
153  if (m_initialized) return ;
154 
155  m_interplator_type = options()->interpolatorType();
156 
157  m_particle_family = mesh()->findItemFamily("Particle");
158  if (m_particle_family)
159  {
160  m_use_particles = options()->useParticles();
161  if (m_use_particles)
162  {
163  MeshCellAccessor* acc = new MeshCellAccessor(mesh());
164  OrthonormalBoundingBox bbox = acc->getBoundingBox();
165  m_epsilon = math::pow(10., -15.) * (bbox.getXMax() - bbox.getXMin()) * (bbox.getYMax() - bbox.getYMin())
166  * (bbox.getZMax() - bbox.getZMin());
167  delete acc ;
168  }
169  }
170  m_initialized = true ;
171 }

301 void AMRDataTransferService::transportFunction(Array<ItemInternal*>& old_items, AMROperationType op)
302 {
303 
304  switch (op)
305  {
306  // Restriction
307  case Restriction: // upscaleDataFunctionP0
308  upscaleDataFunctionP0(old_items);
309  break;
310 
311  // Prolongation
312  case Prolongation: // downScaleFunctionP0
313  downscaleDataFunctionP0(old_items);
314  break;
315  default:
316  throw FatalErrorException(A_FUNCINFO,
317  String::format("No callback function should be called with this operation {1}", op));
318  }
319 
320 }

322 //==========================================================================================================
323 // Upscaling des donn�es d'ordre 0
324 // Dans le cas d'une interpolation d'ordre 0, une seule m�thode suffit pour les variables
325 // scalaires, vectorielles et tensorielles. Pour une interpolation d'ordre plus �lev�, trois
326 // m�thodes distinctes pourraient s'av�rer n�cessaires.
327 //==========================================================================================================
328 void AMRDataTransferService::upscaleDataFunctionP0(Array<ItemInternal*>& old_items)
329 {
330  if (m_use_particles)
331  {
332  std::map<Integer,ItemInternal*> child_to_parent;
333  for (Integer i = 0; i < old_items.size(); i++)
334  {
335  ItemInternal* parent=old_items[i];
336  UInt32 nb_children = parent->nbHChildren() ;
337  for (UInt32 j = 0; j < nb_children; j++)
338  {
339  child_to_parent[Cell(parent).hChild(j).localId()] = parent;
340  }
341  }
342 
343  ENUMERATE_PARTICLE(iparticle,m_particle_family->view())
344  {
345  const Cell& cell = iparticle->cell();
346  std::map<Integer,ItemInternal*>::iterator parent = child_to_parent.find(cell.localId());
347  if (parent != child_to_parent.end())
348  {
349  m_particle_family->toParticleFamily()->setParticleCell(*iparticle, (Cell) (*parent).second);
350  //cout << "Particle upscale:"<<iparticle->localId()<< " from cell :"<<cell.uniqueId()<<" to :"<<((Cell) (*parent).second).uniqueId()<< endl;
351  }
352  }
353  }
354 
355  // Transport de la variable
356  DataToTransferMap::iterator it;
357  for (it = m_data_to_transfer.begin(); it != m_data_to_transfer.end(); ++it)
358  {
359  (*it).second->upscaleP0(old_items) ;
360  }
361 }

363 //==========================================================================================================
364 // Downscaling des donn�es d'ordre 0
365 // Dans le cas d'une interpolation d'ordre 0, une seule m�thode suffit pour les variables
366 // scalaires, vectorielles et tensorielles. Pour une interpolation d'ordre plus �lev�, trois
367 // m�thodes distinctes pourraient s'av�rer n�cessaires.
368 //==========================================================================================================
369 void AMRDataTransferService::downscaleDataFunctionP0(Array<ItemInternal*>& old_items)
370 {
371  if (m_use_particles)
372  {
373 
374  VariableParticleReal3 particle_coords_variable(VariableBuildInfo(mesh(), "ParticleCoord", "Particle"));
375  Real3 particle_coord;
376 
377  m_cell_to_particles.clear();
378  ENUMERATE_PARTICLE(ipart,m_particle_family->view())
379  {
380  const Cell& cell = ipart->cell();
381  particle_coord= particle_coords_variable[ipart];
382  const Integer nb_children = cell.nbHChildren();
383  if (nb_children > 0)
384  {
385  //cout << "Particle downscale:"<<ipart->localId()<< " from cell :"<<cell.uniqueId() ;
386  for (Integer i = 0, is = nb_children; i < is ; i++)
387  {
388  const Cell& child = cell.hChild(i).toCell();
389  if (m_mesh_tools.isInCell(child, particle_coord, m_epsilon))
390  {
391  m_particle_family->toParticleFamily()->setParticleCell(*ipart, child);
392  m_cell_to_particles[child.internal()].addItem(*ipart);
393  //cout <<" to :"<<child.uniqueId()<< endl;
394  break;
395  }
396  }
397  }
398  }
399  }
400 
401  DataToTransferMap::iterator it;
402  for (it = m_data_to_transfer.begin(); it != m_data_to_transfer.end(); ++it)
403  {
404  (*it).second->interpolateP0(old_items) ;
405  }
406 }
