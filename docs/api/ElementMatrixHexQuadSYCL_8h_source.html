51 ARCCORE_HOST_DEVICE inline RealMatrix<8, 8>
52 computeElementMatrixQuad4Base(const RealVector<4>& dxu,
53  const RealVector<4>& dyu,
54  Real integration_weight,
55  Real lambda,
56  Real mu)
57 {
58  // Create displacement gradient vectors for x and y components
59  // For quad4: 4 nodes × 2 DOF = 8 total DOF
60  // Pattern: [u1x, u1y, u2x, u2y, u3x, u3y, u4x, u4y]
61  RealVector<8> dxUx = { dxu(0), 0., dxu(1), 0., dxu(2), 0., dxu(3), 0. };
62  RealVector<8> dyUx = { dyu(0), 0., dyu(1), 0., dyu(2), 0., dyu(3), 0. };
63  RealVector<8> dxUy = { 0., dxu(0), 0., dxu(1), 0., dxu(2), 0., dxu(3) };
64  RealVector<8> dyUy = { 0., dyu(0), 0., dyu(1), 0., dyu(2), 0., dyu(3) };
65 
66  // ∫∫ (λ+2μ)(∂𝑢𝑥/∂𝑥 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑦/∂𝑦)
67  RealMatrix<8, 8> normal_strain_energy = (lambda + 2 * mu) * ((dxUx ^ dxUx) + (dyUy ^ dyUy)) * integration_weight;
68 
69  // ∫∫ λ(∂𝑢𝑦/∂𝑦 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑦/∂𝑦)
70  RealMatrix<8, 8> compressibility_effect = (lambda) * ((dyUy ^ dxUx) + (dxUx ^ dyUy)) * integration_weight;
71 
72  // ∫∫ μ(∂𝑢𝑦/∂𝑥 + ∂𝑢𝑥/∂𝑦)(∂𝑣𝑥/∂𝑦 + ∂𝑣𝑦/∂𝑥)
73  RealMatrix<8, 8> shear_energy = (mu) * ((dxUy + dyUx) ^ (dyUx + dxUy)) * integration_weight;
74 
75  return (normal_strain_energy + compressibility_effect + shear_energy);
76 }

79 ARCCORE_HOST_DEVICE inline RealMatrix<8, 8>
80 computeElementMatrixQuad4Gpu(CellLocalId cell_lid,
81  const IndexedCellNodeConnectivityView& cn_cv,
82  const Accelerator::VariableNodeReal3InView& in_node_coord,
83  const Accelerator::VariableCellRealInView& lambda,
84  const Accelerator::VariableCellRealInView& mu)
85 {
86  // Gauss points and weights for 2x2 quadrature
87  constexpr Real gp[2] = { -M_SQRT1_3, M_SQRT1_3 }; // [-1/sqrt(3) , 1/sqrt(3)]
88  constexpr Real w = 1.0;
89 
90  // Initialize the element matrix
91  RealMatrix<8, 8> ae;
92  ae.fill(0.0);
93 
94  // Loop over Gauss points
95  for (Int8 ixi = 0; ixi < 2; ++ixi)
96  {
97  for (Int8 ieta = 0; ieta < 2; ++ieta) {
98  // Get the coordinates of the Gauss point in natural coordinates (ξ,η)
99  const Real xi = gp[ixi];
100  const Real eta = gp[ieta];
101 
102  // Get shape function gradients w.r.t (𝑥,𝑦) and determinant of Jacobian
103  const auto gp_info =
104  Arcane::FemUtils::Gpu::FeOperation2D::computeGradientsAndJacobianQuad4(cell_lid,
105  cn_cv,
106  in_node_coord,
107  xi,
108  eta);
109  const RealVector<4>& dxU = gp_info.dN_dx;
110  const RealVector<4>& dyU = gp_info.dN_dy;
111  const Real detJ = gp_info.det_j;
112 
113  // Integration weight
114  const Real integration_weight = detJ * w * w;
115 
116  // Add contribution from this Gauss point
117  ae += computeElementMatrixQuad4Base(dxU, dyU, integration_weight, lambda[cell_lid], mu[cell_lid]);
118  }
119  }
120  return ae;
121 }

153 ARCCORE_HOST_DEVICE inline RealMatrix<24, 24>
154 computeElementMatrixHexa8Base(const RealVector<8>& dxu,
155  const RealVector<8>& dyu,
156  const RealVector<8>& dzu,
157  Real integration_weight,
158  Real lambda,
159  Real mu)
160 {
161 
162  // Create displacement gradient vectors for x, y, z components
163  // For hex8: 8 nodes × 3 DOF = 24 total DOF
164  // Pattern: [u1x, u1y, u1z, u2x, u2y, u2z, u3x, u3y, u3z, ..., u8x, u8y, u8z]
165 
166  // X-displacement gradients
167  RealVector<24> dxUx = { dxu(0), 0., 0., dxu(1), 0., 0., dxu(2), 0., 0., dxu(3), 0., 0.,
168  dxu(4), 0., 0., dxu(5), 0., 0., dxu(6), 0., 0., dxu(7), 0., 0. };
169  RealVector<24> dyUx = { dyu(0), 0., 0., dyu(1), 0., 0., dyu(2), 0., 0., dyu(3), 0., 0.,
170  dyu(4), 0., 0., dyu(5), 0., 0., dyu(6), 0., 0., dyu(7), 0., 0. };
171  RealVector<24> dzUx = { dzu(0), 0., 0., dzu(1), 0., 0., dzu(2), 0., 0., dzu(3), 0., 0.,
172  dzu(4), 0., 0., dzu(5), 0., 0., dzu(6), 0., 0., dzu(7), 0., 0. };
173 
174  // Y-displacement gradients
175  RealVector<24> dxUy = { 0., dxu(0), 0., 0., dxu(1), 0., 0., dxu(2), 0., 0., dxu(3), 0.,
176  0., dxu(4), 0., 0., dxu(5), 0., 0., dxu(6), 0., 0., dxu(7), 0. };
177  RealVector<24> dyUy = { 0., dyu(0), 0., 0., dyu(1), 0., 0., dyu(2), 0., 0., dyu(3), 0.,
178  0., dyu(4), 0., 0., dyu(5), 0., 0., dyu(6), 0., 0., dyu(7), 0. };
179  RealVector<24> dzUy = { 0., dzu(0), 0., 0., dzu(1), 0., 0., dzu(2), 0., 0., dzu(3), 0.,
180  0., dzu(4), 0., 0., dzu(5), 0., 0., dzu(6), 0., 0., dzu(7), 0. };
181 
182  // Z-displacement gradients
183  RealVector<24> dxUz = { 0., 0., dxu(0), 0., 0., dxu(1), 0., 0., dxu(2), 0., 0., dxu(3),
184  0., 0., dxu(4), 0., 0., dxu(5), 0., 0., dxu(6), 0., 0., dxu(7) };
185  RealVector<24> dyUz = { 0., 0., dyu(0), 0., 0., dyu(1), 0., 0., dyu(2), 0., 0., dyu(3),
186  0., 0., dyu(4), 0., 0., dyu(5), 0., 0., dyu(6), 0., 0., dyu(7) };
187  RealVector<24> dzUz = { 0., 0., dzu(0), 0., 0., dzu(1), 0., 0., dzu(2), 0., 0., dzu(3),
188  0., 0., dzu(4), 0., 0., dzu(5), 0., 0., dzu(6), 0., 0., dzu(7) };
189 
190  // Normal strain energy terms: ∫∫∫ (λ+2μ)(∂𝑢𝑥/∂𝑥 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑦/∂𝑦 + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑧/∂𝑧)
191  RealMatrix<24, 24> normal_strain_energy = (lambda + 2 * mu) *
192  ((dxUx ^ dxUx) + (dyUy ^ dyUy) + (dzUz ^ dzUz)) * integration_weight;
193 
194  // Compressibility effects: ∫∫∫ λ(∂𝑢𝑦/∂𝑦 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑦/∂𝑦 + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑧/∂𝑧 + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑧/∂𝑧 + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑦/∂𝑦 )
195  RealMatrix<24, 24> compressibility_effect = lambda *
196  ((dxUx ^ (dyUy + dzUz)) + (dyUy ^ (dxUx + dzUz)) + (dzUz ^ (dxUx + dyUy))) * integration_weight;
197 
198  // Shear energy terms: ∫∫∫ μ(∂𝑢𝑦/∂𝑥 + ∂𝑢𝑥/∂𝑦)(∂𝑣𝑥/∂𝑦 + ∂𝑣𝑦/∂𝑥) +
199  // μ(∂𝑢𝑧/∂𝑦 + ∂𝑢𝑦/∂𝑧)(∂𝑣𝑦/∂𝑧 + ∂𝑣𝑧/∂𝑦) +
200  // μ(∂𝑢𝑥/∂𝑧 + ∂𝑢𝑧/∂𝑥)(∂𝑣𝑧/∂𝑥 + ∂𝑣𝑥/∂𝑧)
201  RealMatrix<24, 24> shear_energy = mu *
202  (((dyUx + dxUy) ^ (dyUx + dxUy)) +
203  ((dzUx + dxUz) ^ (dzUx + dxUz)) +
204  ((dzUy + dyUz) ^ (dzUy + dyUz))) * integration_weight;
205 
206  return (normal_strain_energy + compressibility_effect + shear_energy);
207 }

209 ARCCORE_HOST_DEVICE inline RealMatrix<24, 24>
210 computeElementMatrixHexa8Gpu(CellLocalId cell_lid,
211  const IndexedCellNodeConnectivityView& cn_cv,
212  const Accelerator::VariableNodeReal3InView& in_node_coord,
213  const Accelerator::VariableCellRealInView& lambda,
214  const Accelerator::VariableCellRealInView& mu)
215 {
216  // Gauss points and weights for 2x2x2 quadrature
217  constexpr Real gp[2] = { -M_SQRT1_3, M_SQRT1_3 }; // [-1/sqrt(3), 1/sqrt(3)]
218  constexpr Real w = 1.0;
219 
220  // Initialize the element matrix
221  RealMatrix<24, 24> ae;
222  ae.fill(0.0);
223 
224  // Loop over Gauss points
225  for (Int8 ixi = 0; ixi < 2; ++ixi) {
226  for (Int8 ieta = 0; ieta < 2; ++ieta) {
227  for (Int8 izeta = 0; izeta < 2; ++izeta) {
228  // Get the coordinates of the Gauss point in natural coordinates (ξ,η,ζ)
229  const Real xi = gp[ixi];
230  const Real eta = gp[ieta];
231  const Real zeta = gp[izeta];
232 
233  // Get shape function gradients w.r.t (x,y,z) and determinant of Jacobian
234  const auto gp_info =
235  Arcane::FemUtils::Gpu::FeOperation3D::computeGradientsAndJacobianHexa8(
236  cell_lid,cn_cv, in_node_coord, xi, eta, zeta);
237  const RealVector<8>& dxU = gp_info.dN_dx;
238  const RealVector<8>& dyU = gp_info.dN_dy;
239  const RealVector<8>& dzU = gp_info.dN_dz;
240  const Real detJ = gp_info.det_j;
241 
242  // Integration weight
243  const Real integration_weight = detJ * w * w * w;
244 
245  // Add contribution from this Gauss point
246  ae += computeElementMatrixHexa8Base(dxU, dyU, dzU, integration_weight, lambda[cell_lid], mu[cell_lid]);
247  }
248  }
249  }
250  return ae;
251 }
