73 NotMatchingSurfaceImporterService::
74 update(INotMatchingSurfaceMngAccessor * access)
75 {
76  IMesh* mesh = subDomain()->defaultMesh() ;
77 
78 #ifdef USE_ARCANE_V3
79  auto graph = ArcGeoSim::Mesh::GraphMng::graph(mesh,true);
80 #else
81  auto graph = GRAPH(mesh) ;
82 #endif
83  IItemFamily* family = GRAPH(mesh)->linkFamily() ;
84 #ifdef USE_ARCANE_V3
85  m_connectivity = graph->connectivity() ;
86 #endif
87  String name("ContactLinkGroup") ;
88  LinkGroup contact_link_group = family->findGroup(name) ;
89  //LinkGroup contact_link_group = m_subdomain->defaultGraph()->allLinks() ;
90  if(contact_link_group.null())
91  {
92  info()<<"The Not Matching Surface is empty";
93  return;
94  }
95  FaceFaceContactFamily *& face_face_family = access->facefaceContactFamily() ;
96  ENUMERATE_LINK(ilink,contact_link_group)
97  {
98  const Link& link = *ilink ;
99  DualNodeEnumerator inode(DUALNODES_ENUM(link));
100  const Face face1 = DUALITEM(*inode).toFace() ;
101  ++inode ;
102  const Face face2 = DUALITEM(*inode).toFace() ;
103  face_face_family->addContact(face1,face2) ;
104  }
105  face_face_family->endUpdate() ;
106 
107  typedef ContactVariableT<FaceFaceContactFamily::ContactType,Real3> ContactVariableReal3 ;
108  typedef ContactVariableT<FaceFaceContactFamily::ContactType,Real> ContactVariableReal ;
109 
110  ContactVariableReal3& center_var = *new ContactVariableReal3(face_face_family);
111  access->getPropertyVariable(NotMatchingSurfaceProperty::PSimpleCorefinementCenter).reset(&center_var);
112 
113  ContactVariableReal3& normal_var = *new ContactVariableReal3(face_face_family);
114  access->getPropertyVariable(NotMatchingSurfaceProperty::PSimpleCorefinementNormal).reset(&normal_var);
115 
116  ContactVariableReal& vol_var = *new ContactVariableReal(face_face_family);
117  access->getPropertyVariable(NotMatchingSurfaceProperty::PSimpleCorefinementArea).reset(&vol_var);
118 #ifdef USE_ARCANE_V3
119  VariableLinkArrayReal contact_link_properties(VariableBuildInfo(mesh,"MshContactLinkProperties","Links",IVariable::PNoRestore));
120 #else
121  VariableLinkArrayReal contact_link_properties(VariableBuildInfo(mesh,"MshContactLinkProperties",IVariable::PNoRestore));
122 #endif
123  enum eProperties { Vol, Cx, Cy, Cz, Nx, Ny, Nz, NbProperties };
124 
125  //On suppose que iter reste dans l'ordre de declaration des contacts
126  FaceFaceContactEnumerator iter((face_face_family->allContacts()).enumerator());
127  ENUMERATE_LINK(ilink,contact_link_group)
128  {
129  ArrayView<Real> props = contact_link_properties[ilink] ;
130  Real vol = props[Vol] ;
131  Real cx = props[Cx] ;
132  Real cy = props[Cy] ;
133  Real cz = props[Cz] ;
134  Real nx = props[Nx] ;
135  Real ny = props[Ny] ;
136  Real nz = props[Nz] ;
137  vol_var[iter] = Real(vol);
138  normal_var[iter] = Real3(nx,ny,nz) ;
139  center_var[iter] = Real3(cx,cy,cz) ;
140  ++iter ;
141  }
142  //contact_link_properties.resize(0) ;
143 
144  INotMatchingSurfaceMng * mng = access->mng();
145 
146  //if(m_output_level>0)
147  {
148  FaceFaceContactGroup allFaceFaceContacts = mng->facefaceContactFamily()->allContacts();
149 
150  info()<<"---------------------------------------------";
151  info()<<"| Not Matching Surface Info : ";
152  info()<<"| Number of contact objects : "<<allFaceFaceContacts.size();
153  info()<<"---------------------------------------------";
154  info();
155  if(m_output_level>1)
156  {
157  ENUMERATE_FACEFACECONTACT(iter,allFaceFaceContacts)
158  {
159  const FaceFaceContactFamily::ContactType & contact = *iter ;
160  const Cell& iCell = contact.face1().boundaryCell() ;
161  const Cell& jCell = contact.face2().boundaryCell() ;
162  info()<<"Contact : Face A : "<<contact.face1().uniqueId();
163  info()<<" : Face B : "<<contact.face2().uniqueId();
164  info()<<" : Cell A : "<<iCell.uniqueId();
165  info()<<" : Cell B : "<<jCell.uniqueId();
166  }
167  }
168  }
169 }
