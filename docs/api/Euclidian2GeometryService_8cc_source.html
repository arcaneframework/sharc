128 /* Few words about possible optimisations:
129  * Next possible optimizations (when necessary) are:
130  * - Compute common properties on maximal group
131  * (instead of re-computing the same property on all groups with shared items)
132  * - Different properties may use the same computation. Then, maximal group factorisation
133  * may consider by a system of "what properties may compute this kernel" even if not
134  * shared by alls group in a maximal set.
135  * - Mandatory storage parameter in addItemGroupProperty (up to now, this is optional)
136  * - Geometry computations by constraints
137  * ex: for volume computations, we need face center on Quad4. This center is computed twice
138  * for each shared face. Moreover, this computation may be also wished by the used.
139  * The idea is to define requirements for each geometry computations by kind and type.
140  * Then computations are done following the order of the weakest constraints.
141  */

302 void
303 Euclidian2GeometryService::
304 _update2D(CellGroup const& group)
305 {
306  VariableNodeReal3& coords = group.mesh()->nodesCoordinates() ;
307  ENUMERATE_CELL(icell,group)
308  {
309  Real3 center ;
310  for(NodeEnumerator inode(icell->nodes());inode();++inode)
311  {
312  center += coords[inode] ;
313  }
314  center /= icell->nbNode() ;
315  m_cell_center[icell] = Real2(center.x,center.y) ;
316  }
317 
318  Real3 k(0,0,1) ;
319  ENUMERATE_FACE(iface,group.faceGroup())
320  {
321  Real3 center ;
322  for(NodeEnumerator inode(iface->nodes());inode();++inode)
323  {
324  center += coords[inode] ;
325  }
326  center /= iface->nbNode() ;
327  m_face_center[iface] = Real2(center.x,center.y) ;
328  {
329  NodeEnumerator inode(iface->nodes()) ;
330  Real3 u = coords[iface->node(1)] - coords[iface->node(0)] ;
331  Real3 n = math::vecMul(k,u) ;
332  Real2 n2(n.x,n.y) ;
333 
334  // compute normal sign :
335  // sgn = 1 if n is from back cell to front cell
336  // sgn = -1 otherwise
337  Integer sign = 1 ;
338  if(iface->isSubDomainBoundary())
339  {
340  Real2 cf = m_face_center[iface] - m_cell_center[iface->boundaryCell()] ;
341  Real cf_dot_n = math::scaMul(cf,n2) ;
342  if(iface->isSubDomainBoundaryOutside())
343  {
344  if(cf_dot_n<0)
345  sign = -1 ;
346  }
347  else
348  {
349  if(cf_dot_n>0)
350  sign = -1 ;
351  }
352  }
353  else
354  {
355  Real2 cf = m_face_center[iface] - m_cell_center[iface->backCell()] ;
356  Real cf_dot_n = math::scaMul(cf,n2) ;
357  if(cf_dot_n<0)
358  sign = -1 ;
359  }
360  m_face_normal[iface] = Real2(sign*n.x,sign*n.y) ;
361  }
362  }
363 }
