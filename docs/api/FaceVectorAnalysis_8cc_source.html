153  case IT_Quad4:

156  + face_factor*computeNormal(nodes_coordinates, face.node(2), face.node(3), face.node(0));
157  break;
158  case IT_Line2:
159  normal = face_factor * Real3( nodes_coordinates[face.node(1)].z - nodes_coordinates[face.node(0)].z,
160  0,
161  nodes_coordinates[face.node(0)].x - nodes_coordinates[face.node(1)].x );

165  }

167  vx2 += normal.x * normal.x;
168  vy2 += normal.y * normal.y;
169  vz2 += normal.z * normal.z;
170  bbx += normal.y * normal.z;
171  bby += normal.x * normal.z;
172  bbz += normal.x * normal.y;
173  mean_normal += normal;
174  normal_parameters[9] += 1;
175  }
176  }
177
