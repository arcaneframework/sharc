34 inline String ItemGroupBuilder_cleanString(const char * origStr, const bool isUniq) {
35  static int cmpt = 0;
36  const int len = std::strlen(origStr);
37  char * str = new char[len+1];
38  for(int i=0;i<len;++i)
39  {
40  if (std::isalnum(origStr[i]))
41  str[i] = origStr[i];
42  else
43  str[i] = '_';
44  }
45  StringBuilder newString(str,len) ;
46  newString += "_";
47  if (isUniq) newString += cmpt++;
48  delete[] str;
49  return newString.toString() ;
50 }
