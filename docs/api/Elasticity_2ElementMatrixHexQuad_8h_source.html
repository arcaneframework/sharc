50 ARCCORE_HOST_DEVICE inline RealMatrix<8, 8> computeElementMatrixQuad4Base(
51 const RealVector<4>& dxu, const RealVector<4>& dyu, Real integration_weight, Real lambda, Real mu)
52 {
53  // Create displacement gradient vectors for x and y components
54  // For quad4: 4 nodes × 2 DOF = 8 total DOF
55  // Pattern: [u1x, u1y, u2x, u2y, u3x, u3y, u4x, u4y]
56  RealVector<8> dxUx = { dxu(0), 0., dxu(1), 0., dxu(2), 0., dxu(3), 0. };
57  RealVector<8> dyUx = { dyu(0), 0., dyu(1), 0., dyu(2), 0., dyu(3), 0. };
58  RealVector<8> dxUy = { 0., dxu(0), 0., dxu(1), 0., dxu(2), 0., dxu(3) };
59  RealVector<8> dyUy = { 0., dyu(0), 0., dyu(1), 0., dyu(2), 0., dyu(3) };
60 
61  // ∫∫ (λ+2μ)(∂𝑢𝑥/∂𝑥 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑦/∂𝑦)
62  RealMatrix<8, 8> normal_strain_energy = (lambda + 2 * mu) * ((dxUx ^ dxUx) + (dyUy ^ dyUy)) * integration_weight;
63 
64  // ∫∫ λ(∂𝑢𝑦/∂𝑦 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑦/∂𝑦)
65  RealMatrix<8, 8> compressibility_effect = (lambda) * ((dyUy ^ dxUx) + (dxUx ^ dyUy)) * integration_weight;
66 
67  // ∫∫ μ(∂𝑢𝑦/∂𝑥 + ∂𝑢𝑥/∂𝑦)(∂𝑣𝑥/∂𝑦 + ∂𝑣𝑦/∂𝑥)
68  RealMatrix<8, 8> shear_energy = (mu) * ((dxUy + dyUx) ^ (dyUx + dxUy)) * integration_weight;
69 
70  return (normal_strain_energy + compressibility_effect + shear_energy);
71 }

73 RealMatrix<8, 8> FemModuleElasticity::_computeElementMatrixQuad4(Cell cell)
74 {
75  // Gauss points and weights for 2x2 quadrature
76  constexpr Real gp[2] = { -M_SQRT1_3, M_SQRT1_3 }; // [-1/sqrt(3) , 1/sqrt(3)]
77  constexpr Real w = 1.0;
78 
79  // Initialize the element matrix
80  RealMatrix<8, 8> ae;
81  ae.fill(0.0);
82 
83  // Loop over Gauss points
84  for (Int8 ixi = 0; ixi < 2; ++ixi) {
85  for (Int8 ieta = 0; ieta < 2; ++ieta) {
86  // Get the coordinates of the Gauss point in natural coordinates (ξ,η)
87  const Real xi = gp[ixi];
88  const Real eta = gp[ieta];
89 
90  // Get shape function gradients w.r.t (𝑥,𝑦) and determinant of Jacobian
91  const auto gp_info = ArcaneFemFunctions::FeOperation2D::computeGradientsAndJacobianQuad4(cell, m_node_coord, xi, eta);
92  const RealVector<4>& dxU = gp_info.dN_dx;
93  const RealVector<4>& dyU = gp_info.dN_dy;
94  const Real detJ = gp_info.det_j;
95 
96  // Integration weight
97  const Real integration_weight = detJ * w * w;
98 
99  // Add contribution from this Gauss point
100  ae += computeElementMatrixQuad4Base(dxU, dyU, integration_weight, m_cell_lambda[cell], m_cell_mu[cell]);
101  }
102  }
103 
104  return ae;
105 }

179 ARCCORE_HOST_DEVICE RealMatrix<24, 24> computeElementMatrixHexa8Base(
180 const RealVector<8>& dxu, const RealVector<8>& dyu, const RealVector<8>& dzu,
181 Real integration_weight, Real lambda, Real mu)
182 {
183 
184  // Create displacement gradient vectors for x, y, z components
185  // For hex8: 8 nodes × 3 DOF = 24 total DOF
186  // Pattern: [u1x, u1y, u1z, u2x, u2y, u2z, u3x, u3y, u3z, ..., u8x, u8y, u8z]
187 
188  // X-displacement gradients
189  RealVector<24> dxUx = { dxu(0), 0., 0., dxu(1), 0., 0., dxu(2), 0., 0., dxu(3), 0., 0.,
190  dxu(4), 0., 0., dxu(5), 0., 0., dxu(6), 0., 0., dxu(7), 0., 0. };
191  RealVector<24> dyUx = { dyu(0), 0., 0., dyu(1), 0., 0., dyu(2), 0., 0., dyu(3), 0., 0.,
192  dyu(4), 0., 0., dyu(5), 0., 0., dyu(6), 0., 0., dyu(7), 0., 0. };
193  RealVector<24> dzUx = { dzu(0), 0., 0., dzu(1), 0., 0., dzu(2), 0., 0., dzu(3), 0., 0.,
194  dzu(4), 0., 0., dzu(5), 0., 0., dzu(6), 0., 0., dzu(7), 0., 0. };
195 
196  // Y-displacement gradients
197  RealVector<24> dxUy = { 0., dxu(0), 0., 0., dxu(1), 0., 0., dxu(2), 0., 0., dxu(3), 0.,
198  0., dxu(4), 0., 0., dxu(5), 0., 0., dxu(6), 0., 0., dxu(7), 0. };
199  RealVector<24> dyUy = { 0., dyu(0), 0., 0., dyu(1), 0., 0., dyu(2), 0., 0., dyu(3), 0.,
200  0., dyu(4), 0., 0., dyu(5), 0., 0., dyu(6), 0., 0., dyu(7), 0. };
201  RealVector<24> dzUy = { 0., dzu(0), 0., 0., dzu(1), 0., 0., dzu(2), 0., 0., dzu(3), 0.,
202  0., dzu(4), 0., 0., dzu(5), 0., 0., dzu(6), 0., 0., dzu(7), 0. };
203 
204  // Z-displacement gradients
205  RealVector<24> dxUz = { 0., 0., dxu(0), 0., 0., dxu(1), 0., 0., dxu(2), 0., 0., dxu(3),
206  0., 0., dxu(4), 0., 0., dxu(5), 0., 0., dxu(6), 0., 0., dxu(7) };
207  RealVector<24> dyUz = { 0., 0., dyu(0), 0., 0., dyu(1), 0., 0., dyu(2), 0., 0., dyu(3),
208  0., 0., dyu(4), 0., 0., dyu(5), 0., 0., dyu(6), 0., 0., dyu(7) };
209  RealVector<24> dzUz = { 0., 0., dzu(0), 0., 0., dzu(1), 0., 0., dzu(2), 0., 0., dzu(3),
210  0., 0., dzu(4), 0., 0., dzu(5), 0., 0., dzu(6), 0., 0., dzu(7) };
211 
212  // Normal strain energy terms: ∫∫∫ (λ+2μ)(∂𝑢𝑥/∂𝑥 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑦/∂𝑦 + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑧/∂𝑧)
213  RealMatrix<24, 24> normal_strain_energy = (lambda + 2 * mu) *
214  ((dxUx ^ dxUx) + (dyUy ^ dyUy) + (dzUz ^ dzUz)) * integration_weight;
215 
216  // Compressibility effects: ∫∫∫ λ(∂𝑢𝑦/∂𝑦 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑦/∂𝑦 + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑧/∂𝑧 + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑧/∂𝑧 + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑦/∂𝑦 )
217  RealMatrix<24, 24> compressibility_effect = lambda *
218  ((dxUx ^ (dyUy + dzUz)) + (dyUy ^ (dxUx + dzUz)) + (dzUz ^ (dxUx + dyUy))) * integration_weight;
219 
220  // Shear energy terms: ∫∫∫ μ(∂𝑢𝑦/∂𝑥 + ∂𝑢𝑥/∂𝑦)(∂𝑣𝑥/∂𝑦 + ∂𝑣𝑦/∂𝑥) +
221  // μ(∂𝑢𝑧/∂𝑦 + ∂𝑢𝑦/∂𝑧)(∂𝑣𝑦/∂𝑧 + ∂𝑣𝑧/∂𝑦) +
222  // μ(∂𝑢𝑥/∂𝑧 + ∂𝑢𝑧/∂𝑥)(∂𝑣𝑧/∂𝑥 + ∂𝑣𝑥/∂𝑧)
223  RealMatrix<24, 24> shear_energy = mu *
224  (((dyUx + dxUy) ^ (dyUx + dxUy)) +
225  ((dzUx + dxUz) ^ (dzUx + dxUz)) +
226  ((dzUy + dyUz) ^ (dzUy + dyUz))) * integration_weight;
227 
228  return (normal_strain_energy + compressibility_effect + shear_energy);
229 }
