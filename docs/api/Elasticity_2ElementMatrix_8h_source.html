41 ARCCORE_HOST_DEVICE RealMatrix<6, 6> computeElementMatrixTria3Base(Real3 dxu, Real3 dyu, Real area, Real lambda, Real mu)
42 {
43  RealVector<6> dxUx = { dxu[0], 0., dxu[1], 0., dxu[2], 0. };
44  RealVector<6> dyUx = { dyu[0], 0., dyu[1], 0., dyu[2], 0. };
45  RealVector<6> dxUy = { 0., dxu[0], 0., dxu[1], 0., dxu[2] };
46  RealVector<6> dyUy = { 0., dyu[0], 0., dyu[1], 0., dyu[2] };
47 
48  // ∫∫ (λ+2μ)(∂𝑢𝑥/∂𝑥 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑦/∂𝑦)
49  RealMatrix<6, 6> normal_strain_energy = (lambda + 2 * mu) * ((dxUx ^ dxUx) + (dyUy ^ dyUy)) * area;
50 
51  // ∫∫ λ(∂𝑢𝑦/∂𝑦 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑦/∂𝑦)
52  RealMatrix<6, 6> compressibility_effect = (lambda) * ((dyUy ^ dxUx) + (dxUx ^ dyUy)) * area;
53 
54  // ∫∫ μ(∂𝑢𝑦/∂𝑥 + ∂𝑢𝑥/∂𝑦)(∂𝑣𝑥/∂𝑦 + ∂𝑣𝑦/∂𝑥)
55  RealMatrix<6, 6> shear_energy = (mu) * ((dxUy + dyUx) ^ (dyUx + dxUy)) * area;
56 
57  return (normal_strain_energy + compressibility_effect + shear_energy);
58 }

92 ARCCORE_HOST_DEVICE RealMatrix<2, 6>
93 computeElementVectorTria3Gpu(CellLocalId cell_lid,
94  const IndexedCellNodeConnectivityView& cn_cv,
95  const Accelerator::VariableNodeReal3InView& in_node_coord,
96  const Accelerator::VariableCellRealInView& lambda,
97  const Accelerator::VariableCellRealInView& mu,
98  Int32 node_lid)
99 {
100  Real3 dxu = Arcane::FemUtils::Gpu::FeOperation2D::computeGradientXTria3(cell_lid, cn_cv, in_node_coord);
101  Real3 dyu = Arcane::FemUtils::Gpu::FeOperation2D::computeGradientYTria3(cell_lid, cn_cv, in_node_coord);
102  Real area = Arcane::FemUtils::Gpu::MeshOperation::computeAreaTria3(cell_lid, cn_cv, in_node_coord);
103 
104  RealVector<6> dxUx = { dxu[0], 0., dxu[1], 0., dxu[2], 0. };
105  RealVector<6> dyUx = { dyu[0], 0., dyu[1], 0., dyu[2], 0. };
106  RealVector<6> dxUy = { 0., dxu[0], 0., dxu[1], 0., dxu[2] };
107  RealVector<6> dyUy = { 0., dyu[0], 0., dyu[1], 0., dyu[2] };
108 
109  // ∫∫ (λ+2μ)(∂𝑢𝑥/∂𝑥 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑦/∂𝑦)
110  RealVector<6> normal_strain_energy_x = (lambda[cell_lid] + 2 * mu[cell_lid]) * ((dxUx(node_lid*2) * dxUx) + (dyUy(node_lid*2) * dyUy)) * area;
111  RealVector<6> normal_strain_energy_y = (lambda[cell_lid] + 2 * mu[cell_lid]) * ((dxUx(node_lid*2+1) * dxUx) + (dyUy(node_lid*2+1) * dyUy)) * area;
112 
113  // ∫∫ λ(∂𝑢𝑦/∂𝑦 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑦/∂𝑦)
114  RealVector<6> compressibility_effect_x = (lambda[cell_lid]) * ((dyUy(node_lid*2) * dxUx) + (dxUx(node_lid*2) * dyUy)) * area;
115  RealVector<6> compressibility_effect_y = (lambda[cell_lid]) * ((dyUy(node_lid*2+1) * dxUx) + (dxUx(node_lid*2+1) * dyUy)) * area;
116 
117  // ∫∫ μ(∂𝑢𝑦/∂𝑥 + ∂𝑢𝑥/∂𝑦)(∂𝑣𝑥/∂𝑦 + ∂𝑣𝑦/∂𝑥)
118  RealVector<6> shear_energy_x = (mu[cell_lid]) * ((dxUy(node_lid*2) + dyUx(node_lid*2)) * (dyUx + dxUy)) * area;
119  RealVector<6> shear_energy_y = (mu[cell_lid]) * ((dxUy(node_lid*2+1) + dyUx(node_lid*2+1)) * (dyUx + dxUy)) * area;
120 
121  RealVector <6> result_x = normal_strain_energy_x + compressibility_effect_x + shear_energy_x;
122  RealVector <6> result_y = normal_strain_energy_y + compressibility_effect_y + shear_energy_y;
123 
124  RealMatrix<2, 6> result = {
125  { result_x(0), result_x(1), result_x(2), result_x(3), result_x(4), result_x(5) },
126  { result_y(0), result_y(1), result_y(2), result_y(3), result_y(4), result_y(5) }
127  };
128 
129  return result;
130 }

162 ARCCORE_HOST_DEVICE RealMatrix<12, 12> computeElementMatrixTetra4Base(Real4 dxu, Real4 dyu, Real4 dzu, Real volume, Real lambda, Real mu)
163 {
164  RealVector<12> dxUx = { dxu[0], 0., 0., dxu[1], 0., 0., dxu[2], 0., 0., dxu[3], 0., 0. };
165  RealVector<12> dyUx = { dyu[0], 0., 0., dyu[1], 0., 0., dyu[2], 0., 0., dyu[3], 0., 0. };
166  RealVector<12> dzUx = { dzu[0], 0., 0., dzu[1], 0., 0., dzu[2], 0., 0., dzu[3], 0., 0. };
167 
168  RealVector<12> dxUy = { 0., dxu[0], 0., 0., dxu[1], 0., 0., dxu[2], 0., 0., dxu[3], 0. };
169  RealVector<12> dyUy = { 0., dyu[0], 0., 0., dyu[1], 0., 0., dyu[2], 0., 0., dyu[3], 0. };
170  RealVector<12> dzUy = { 0., dzu[0], 0., 0., dzu[1], 0., 0., dzu[2], 0., 0., dzu[3], 0. };
171 
172  RealVector<12> dxUz = { 0., 0., dxu[0], 0., 0., dxu[1], 0., 0., dxu[2], 0., 0., dxu[3] };
173  RealVector<12> dyUz = { 0., 0., dyu[0], 0., 0., dyu[1], 0., 0., dyu[2], 0., 0., dyu[3] };
174  RealVector<12> dzUz = { 0., 0., dzu[0], 0., 0., dzu[1], 0., 0., dzu[2], 0., 0., dzu[3] };
175 
176  // ∫∫∫ (λ+2μ)(∂𝑢𝑥/∂𝑥 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑦/∂𝑦 + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑧/∂𝑧)
177  RealMatrix<12, 12> normal_strain_energy = (lambda + 2 * mu) * ((dxUx ^ dxUx) + (dyUy ^ dyUy) + (dzUz ^ dzUz) ) * volume;
178 
179  // ∫∫∫ λ(∂𝑢𝑦/∂𝑦 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑦/∂𝑦
180  // + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑧/∂𝑧
181  // + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑧/∂𝑧 + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑦/∂𝑦)
182  RealMatrix<12, 12> compressibility_effect = (lambda) * ((dyUy ^ dxUx) + (dxUx ^ dyUy) +
183  (dzUz ^ dxUx) + (dxUx ^ dzUz) +
184  (dyUy ^ dzUz) + (dzUz ^ dyUy) ) * volume;
185 
186  // ∫∫∫ μ(∂𝑢𝑦/∂𝑥 + ∂𝑢𝑥/∂𝑦)(∂𝑣𝑥/∂𝑦 + ∂𝑣𝑦/∂𝑥) +
187  // μ(∂𝑢𝑧/∂𝑦 + ∂𝑢𝑦/∂𝑧)(∂𝑣𝑦/∂𝑧 + ∂𝑣𝑧/∂𝑦) +
188  // μ(∂𝑢𝑥/∂𝑧 + ∂𝑢𝑧/∂𝑥)(∂𝑣𝑧/∂𝑥 + ∂𝑣𝑥/∂𝑧)
189  RealMatrix<12, 12> shear_energy = (mu) * ( ((dxUy + dyUx) ^ (dyUx + dxUy)) +
190  ((dzUy + dyUz) ^ (dyUz + dzUy)) +
191  ((dxUz + dzUx) ^ (dzUx + dxUz)) ) * volume;
192 
193  return (normal_strain_energy + compressibility_effect + shear_energy);
194 }

235  const Accelerator::VariableNodeReal3InView& in_node_coord,
236  const Accelerator::VariableCellRealInView& lambda,
237  const Accelerator::VariableCellRealInView& mu,
238  Int32 node_lid)
239 {
240  Real4 dxu = Arcane::FemUtils::Gpu::FeOperation3D::computeGradientXTetra4(cell_lid, cn_cv, in_node_coord);
241  Real4 dyu = Arcane::FemUtils::Gpu::FeOperation3D::computeGradientYTetra4(cell_lid, cn_cv, in_node_coord);
242  Real4 dzu = Arcane::FemUtils::Gpu::FeOperation3D::computeGradientZTetra4(cell_lid, cn_cv, in_node_coord);
243 
244  Real volume = Arcane::FemUtils::Gpu::MeshOperation::computeVolumeTetra4(cell_lid, cn_cv, in_node_coord);
245 
246  RealVector<12> dxUx = { dxu[0], 0., 0., dxu[1], 0., 0., dxu[2], 0., 0., dxu[3], 0., 0. };
247  RealVector<12> dyUx = { dyu[0], 0., 0., dyu[1], 0., 0., dyu[2], 0., 0., dyu[3], 0., 0. };
248  RealVector<12> dzUx = { dzu[0], 0., 0., dzu[1], 0., 0., dzu[2], 0., 0., dzu[3], 0., 0. };
249 
250  RealVector<12> dxUy = { 0., dxu[0], 0., 0., dxu[1], 0., 0., dxu[2], 0., 0., dxu[3], 0. };
251  RealVector<12> dyUy = { 0., dyu[0], 0., 0., dyu[1], 0., 0., dyu[2], 0., 0., dyu[3], 0. };
252  RealVector<12> dzUy = { 0., dzu[0], 0., 0., dzu[1], 0., 0., dzu[2], 0., 0., dzu[3], 0. };
253 
254  RealVector<12> dxUz = { 0., 0., dxu[0], 0., 0., dxu[1], 0., 0., dxu[2], 0., 0., dxu[3] };
255  RealVector<12> dyUz = { 0., 0., dyu[0], 0., 0., dyu[1], 0., 0., dyu[2], 0., 0., dyu[3] };
256  RealVector<12> dzUz = { 0., 0., dzu[0], 0., 0., dzu[1], 0., 0., dzu[2], 0., 0., dzu[3] };
257 
258  // ∫∫∫ (λ+2μ)(∂𝑢𝑥/∂𝑥 ∂𝑣𝑥/∂𝑥 +
259  // ∂𝑢𝑦/∂𝑦 ∂𝑣𝑦/∂𝑦 +
260  // ∂𝑢𝑧/∂𝑧 ∂𝑣𝑧/∂𝑧 )
261  RealVector<12> normal_strain_energy_x = (lambda[cell_lid] + 2 * mu[cell_lid]) * ( (dxUx(node_lid*3) * dxUx) +
262  (dyUy(node_lid*3) * dyUy) +
263  (dzUz(node_lid*3) * dzUz) ) * volume;
264 
265  RealVector<12> normal_strain_energy_y = (lambda[cell_lid] + 2 * mu[cell_lid]) * ( (dxUx(node_lid*3+1) * dxUx) +
266  (dyUy(node_lid*3+1) * dyUy) +
267  (dzUz(node_lid*3+1) * dzUz) ) * volume;
268 
269  RealVector<12> normal_strain_energy_z = (lambda[cell_lid] + 2 * mu[cell_lid]) * ( (dxUx(node_lid*3+2) * dxUx) +
270  (dyUy(node_lid*3+2) * dyUy) +
271  (dzUz(node_lid*3+2) * dzUz) ) * volume;
272 
273  // ∫∫∫ λ(∂𝑢𝑦/∂𝑦 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑦/∂𝑦
274  // + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑧/∂𝑧
275  // + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑧/∂𝑧 + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑦/∂𝑦)
276  RealVector<12> compressibility_effect_x = (lambda[cell_lid]) * ((dyUy(node_lid*3) * dxUx) + (dxUx(node_lid*3) * dyUy) +
277  (dzUz(node_lid*3) * dxUx) + (dxUx(node_lid*3) * dzUz) +
278  (dyUy(node_lid*3) * dzUz) + (dzUz(node_lid*3) * dyUy) ) * volume;
279 
280  RealVector<12> compressibility_effect_y = (lambda[cell_lid]) * ((dyUy(node_lid*3+1) * dxUx) + (dxUx(node_lid*3+1) * dyUy) +
281  (dzUz(node_lid*3+1) * dxUx) + (dxUx(node_lid*3+1) * dzUz) +
282  (dyUy(node_lid*3+1) * dzUz) + (dzUz(node_lid*3+1) * dyUy) ) * volume;
283 
284  RealVector<12> compressibility_effect_z = (lambda[cell_lid]) * ((dyUy(node_lid*3+2) * dxUx) + (dxUx(node_lid*3+2) * dyUy) +
285  (dzUz(node_lid*3+2) * dxUx) + (dxUx(node_lid*3+2) * dzUz) +
286  (dyUy(node_lid*3+2) * dzUz) + (dzUz(node_lid*3+2) * dyUy) ) * volume;
287 
288 
289  // ∫∫∫ μ(∂𝑢𝑦/∂𝑥 + ∂𝑢𝑥/∂𝑦)(∂𝑣𝑥/∂𝑦 + ∂𝑣𝑦/∂𝑥) +
290  // μ(∂𝑢𝑧/∂𝑦 + ∂𝑢𝑦/∂𝑧)(∂𝑣𝑦/∂𝑧 + ∂𝑣𝑧/∂𝑦) +
291  // μ(∂𝑢𝑥/∂𝑧 + ∂𝑢𝑧/∂𝑥)(∂𝑣𝑧/∂𝑥 + ∂𝑣𝑥/∂𝑧)
292  RealVector<12> shear_energy_x = (mu[cell_lid]) * ( ((dxUy(node_lid*3) + dyUx(node_lid*3)) * (dyUx + dxUy)) +
293  ((dzUy(node_lid*3) + dyUz(node_lid*3)) * (dyUz + dzUy)) +
294  ((dxUz(node_lid*3) + dzUx(node_lid*3)) * (dzUx + dxUz)) ) * volume;
295 
296  RealVector<12> shear_energy_y = (mu[cell_lid]) * ( ((dxUy(node_lid*3+1) + dyUx(node_lid*3+1)) * (dyUx + dxUy)) +
297  ((dzUy(node_lid*3+1) + dyUz(node_lid*3+1)) * (dyUz + dzUy)) +
298  ((dxUz(node_lid*3+1) + dzUx(node_lid*3+1)) * (dzUx + dxUz)) ) * volume;
299 
300 
301 
302  RealVector<12> shear_energy_z = (mu[cell_lid]) * ( ((dxUy(node_lid*3+2) + dyUx(node_lid*3+2)) * (dyUx + dxUy)) +
303  ((dzUy(node_lid*3+2) + dyUz(node_lid*3+2)) * (dyUz + dzUy)) +
304  ((dxUz(node_lid*3+2) + dzUx(node_lid*3+2)) * (dzUx + dxUz)) ) * volume;
305 
306 
307  RealVector<12> result_x = normal_strain_energy_x + compressibility_effect_x + shear_energy_x;
308  RealVector<12> result_y = normal_strain_energy_y + compressibility_effect_y + shear_energy_y;
309  RealVector<12> result_z = normal_strain_energy_z + compressibility_effect_z + shear_energy_z;
310 
311  RealMatrix<3, 12> result = {
312  { result_x(0), result_x(1), result_x(2), result_x(3), result_x(4), result_x(5),
313  result_x(6), result_x(7), result_x(8), result_x(9), result_x(10), result_x(11) },
314 
315  { result_y(0), result_y(1), result_y(2), result_y(3), result_y(4), result_y(5),
316  result_y(6), result_y(7), result_y(8), result_y(9), result_y(10), result_y(11) },
317 
318  { result_z(0), result_z(1), result_z(2), result_z(3), result_z(4), result_z(5),
319  result_z(6), result_z(7), result_z(8), result_z(9), result_z(10), result_z(11) }
320  };
321 
322  return result;
