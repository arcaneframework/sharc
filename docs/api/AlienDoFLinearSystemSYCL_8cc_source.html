113 void AlienDoFLinearSystemImpl::
114 _setVectorsAcc(Alien::Vector& vectorX, Alien::Vector& vectorB)
115 {
116  IItemFamily* dof_family = dofFamily();
117  VariableDoFReal& rhs_variable = this->rhsVariable();
118  VariableDoFReal& dof_variable = this->solutionVariable();
119  Arccore::SmallSpan<const Int32> dof_lids = dof_family->allItems().own().view().localIds() ;
120  Arccore::SmallSpan<const Int32> allUIndex = m_alien_bsr_format->getAllUIndex() ;
121 
122  {
123  auto vx_acc = Alien::SYCL::VectorAccessorT<Real>(vectorX);
124  auto vb_acc = Alien::SYCL::VectorAccessorT<Real>(vectorB);
125 
126  //sycl::buffer<Integer,1> dof_lids_buffer(dof_lids.data(),sycl::range(dof_lids.size())) ;
127  sycl::buffer<Integer,1> allUIndex_buffer(allUIndex.data(),sycl::range(allUIndex.size())) ;
128 
129  Alien::ParallelEngine engine(m_alien_bsr_format->queue()) ;
130 
131  engine.submit([&](ControlGroupHandler& handler)
132  {
133  auto& command = handler.command() ;
134  auto in_dof_lids = ax::viewIn(command,dof_lids);
135  //auto in_dof_lids = dof_lids_buffer.get_access<sycl::access::mode::read>(handler.m_internal) ;
136  //auto in_allUIndex = ax::viewIn(command,accAllUIndex) ;
137  auto in_allUIndex = allUIndex_buffer.get_access<sycl::access::mode::read>(handler.m_internal) ;
138 
139  auto in_result = ax::viewIn(command,dof_variable);
140  auto in_rhs = ax::viewIn(command,rhs_variable);
141 
142  auto out_vx = vx_acc.view(handler) ;
143  auto out_vb = vb_acc.view(handler) ;
144 
145  auto local_size = dof_lids.size() ;
146  handler.parallel_for(engine.maxNumThreads(),
147  [=](Alien::ParallelEngine::Item<1>::type item)
148  {
149  auto id = item.get_id(0) ;
150  for (auto index = id; index < local_size; index += item.get_range()[0])
151  {
152  auto vi = DoFLocalId(in_dof_lids[index]) ;
153  auto lid = in_dof_lids[index] ;
154  auto iIndex = in_allUIndex[lid];
155  if(iIndex!=-1)
156  {
157  out_vx[iIndex] = in_result[vi] ;
158  out_vb[iIndex] = in_rhs[vi] ;
159  }
160  } ;
161  }) ;
162 
163  }) ;
164  }

167 void AlienDoFLinearSystemImpl::
168 _getSolutionAcc(Alien::Vector& vectorX)
169 {
170  IItemFamily* dof_family = dofFamily();
171  VariableDoFReal& rhs_variable = this->rhsVariable();
172  VariableDoFReal& dof_variable = this->solutionVariable();
173  Arccore::SmallSpan<const Int32> dof_lids = dof_family->allItems().own().view().localIds() ;
174 
175  Arccore::SmallSpan<const Int32> allUIndex = m_alien_bsr_format->getAllUIndex() ;
176  {
177  sycl::buffer<Integer,1> allUIndex_buffer(allUIndex.data(),sycl::range(allUIndex.size())) ;
178  auto const& sycl_X = vectorX.impl()->get<Alien::BackEnd::tag::sycl>() ;
179  auto x_buffer = sycl_X.internal()->values() ;
180  Alien::ParallelEngine engine(m_alien_bsr_format->queue()) ;
181 
182  engine.submit([&](ControlGroupHandler& handler)
183  {
184  auto& command = handler.command() ;
185  auto in_dof_lids = ax::viewIn(command,dof_lids);
186  //auto in_allUIndex = ax::viewIn(command,accAllUIndex) ;
187  auto in_allUIndex = allUIndex_buffer.get_access<sycl::access::mode::read>(handler.m_internal) ;
188  auto out_result = ax::viewOut(command,dof_variable);
189  auto in_vx = x_buffer.get_access<sycl::access::mode::read>(handler.m_internal) ;
190 
191  auto local_size = dof_lids.size() ;
192  handler.parallel_for(engine.maxNumThreads(),

194  {
195  auto id = item.get_id(0) ;
196  for (auto index = id; index < local_size; index += item.get_range()[0])
197  {
198  auto vi = DoFLocalId(in_dof_lids[index]) ;
199  auto lid = in_dof_lids[index] ;
200  auto iIndex = in_allUIndex[lid];
201  if(iIndex!=-1)
202  {
203  out_result[vi] = in_vx[iIndex];
204  }
205  } ;
206  }) ;
207 
208  }) ;
209  }

212 void AlienDoFLinearSystemImpl::
213 _applyForcedValuesToLhsAcc()
214 {
215  IItemFamily* dof_family = dofFamily();
216  VariableDoFBool& is_forced = this->getForcedInfo() ;
217  VariableDoFReal& forced_value = this->getForcedValue() ;
218 
219  auto& matrixA = m_alien_bsr_format->getMatrixA();
220 
221  Arccore::SmallSpan<const Int32> dof_lids = dof_family->allItems().own().view().localIds() ;
222  Arccore::SmallSpan<const Integer> allUIndex = m_alien_bsr_format->getAllUIndex() ;
223  {
224 
225  Alien::SYCL::ProfiledMatrixBuilder builder(matrixA, Alien::ProfiledMatrixOptions::eKeepValues);
226  Alien::ParallelEngine engine(m_alien_bsr_format->queue()) ;
227 
228  sycl::buffer<Integer,1> allUIndex_buffer(allUIndex.data(),sycl::range(allUIndex.size())) ;
229 
230  engine.submit([&](ControlGroupHandler& handler)
231  {
232  auto& command = handler.command() ;
233  auto in_dof_lids = ax::viewIn(command,dof_lids);
234  //auto in_allUIndex = ax::viewIn(command,allUIndex) ;
235  auto in_allUIndex = allUIndex_buffer.get_access<sycl::access::mode::read>(handler.m_internal) ;
236  auto in_is_forced = ax::viewIn(command,is_forced);
237  auto in_forced_value = ax::viewIn(command,forced_value);
238  auto matrix_acc = builder.view(handler) ;
239  auto local_size = dof_lids.size() ;
240  handler.parallel_for(engine.maxNumThreads(),
241  [=](Alien::ParallelEngine::Item<1>::type item)
242  {
243  auto id = item.get_id(0) ;
244  for (auto index = id; index < local_size; index += item.get_range()[0])
245  {
246  auto vi = DoFLocalId(in_dof_lids[index]) ;
247  auto lid = in_dof_lids[index] ;
248  auto row_i = in_allUIndex[lid];
249  auto forced = in_is_forced[vi] ;
250  Real value = in_forced_value[vi];
251  if(row_i!=-1 && forced)
252  {
253  auto eii = matrix_acc.entryIndex(row_i, row_i) ;
254  matrix_acc[eii] = value;
255  }
256  } ;
257  }) ;
258 
259  }) ;
260  }

264 void AlienDoFLinearSystemImpl::
265 _printMatrixAcc(Alien::Matrix& matrix)
266 {
267 
268  auto const& true_A = matrix.impl()->get<Alien::BackEnd::tag::hcsr>() ;
269 
270  auto nrows = true_A.getLocalSize();
271  auto kcol_buffer = true_A.internal()->kcol() ;
272  auto cols_buffer = true_A.internal()->cols() ;
273  auto values_buffer = true_A.internal()->values() ;
274  sycl::host_accessor<Real, 1, sycl::access::mode::read> values(values_buffer);
275  sycl::host_accessor<int, 1, sycl::access::mode::read> kcol(kcol_buffer);
276  sycl::host_accessor<int, 1, sycl::access::mode::read> cols(cols_buffer);
277 
278  for(int irow=0;irow<nrows;++irow)
279  {
280  std::cout<<"ROW["<<irow<<"] : ";
281  for(int k=kcol[irow];k<kcol[irow+1];++k)
282  {
283  std::cout<<"("<<cols[k]<<","<<values[k]<<") ";
284  }
285  std::cout<<std::endl;
286  }
